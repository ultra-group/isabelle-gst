lemmas Subset_def = spec[OF spec[OF Subset_ax]]
   and SetMem_def = spec[OF SetMem_ax]
   and ReplPred_def = spec[OF spec[OF ReplPred_ax]]
   and SetOf_def = spec[OF spec[OF SetOf_ax]]

subsection ‹Soft typing rules›

declare SetOf_def [typdef] SetMem_def [typdef]

lemma setof_set_subtyp : "SetOf α << Set"
  unfolding SetOf_def by (rule subtypI, drule tyE, simp)

lemma setof_iff :
  "x : SetOf α ⟷ x : Set ∧ (∀b. b ∈ x ⟶ b : α)"
  unfolding SetOf_def by (unfold_typs) 

lemma setof_subtyp : "α << β ⟹ SetOf α << SetOf β"
  unfolding subtyp_def setof_iff by auto

lemma setofI : "⟦ x : Set ; ⋀b. b ∈ x ⟹ b : P ⟧ ⟹ x : SetOf P"
  unfolding SetOf_def by (unfold_typs, auto)

lemma setof_mem : "x : SetOf α ⟹ b ∈ x ⟹ b : α"
  unfolding SetOf_def has_ty_def by auto 

lemma setmemI : "x : Set ⟹ a ∈ x ⟹ a : SetMem"
  unfolding SetMem_def by (unfold_typs, auto)

lemma setmem_iff :
 "b : SetMem ⟷ (∃x : Set. b ∈ x)" 
 by (unfold_typs)

lemma setof_set : "x : SetOf α ⟹ x : Set"
  by (rule subtypE[OF setof_set_subtyp])

lemma set_setof : "x : Set ⟹ x : SetOf (SetMem)"
  by (rule setofI, use setmemI in auto)  

definition MemOf :: "'a ⇒ 'a ⇒ bool"
  where "MemOf x b ≡ b ∈ x"

lemma memofI :
  assumes "b ∈ x"
  shows "b : MemOf x"
  using assms unfolding MemOf_def
  by (unfold_typs)

lemma memofD : 
  assumes "b : MemOf x"
  shows "b ∈ x"
  using assms unfolding MemOf_def 
  by (unfold_typs)

lemmas memof_setmem = setmemI[OF _ memofD]

lemma set_setof_memof :
  assumes x:"x : Set"
  shows "x : SetOf (MemOf x)"
  using setofI[OF x memofI] by auto

abbreviation not_mem :: "['a, 'a] ⇒ bool"  (infixl ‹∉› 50)  ― ‹negated membership relation›
  where "x ∉ y ≡ ¬ (x ∈ y)"

section ‹Bounding quantification over a set›

definition ball :: "'a ⇒ ('a ⇒ bool) ⇒ bool" where
  ball_def : "ball x P ≡ rall (λa. a ∈ x) P" 
definition bex :: "'a ⇒ ('a ⇒ bool) ⇒ bool" where
  bex_def : "bex x P ≡ rex (λa. a ∈ x) P"

end

(*Temporarily exiting GZF context to perform
  syntax declarations for ∈-bounded quantification*)
syntax
  "_ball" :: "[pttrn, 'a, bool] ⇒ bool"  (‹(3∀_∈_./ _)› 10)
  "_bex"  :: "[pttrn, 'a, bool] ⇒ bool"   (‹(3∃_∈_./ _)› 10)
translations
  "∀x∈A. P" ⇌ "CONST ball A (λx. P)"
  "∃x∈A. P" ⇌ "CONST bex A (λx. P)"

syntax
  "_depfun_set" :: "[pttrn, 'a, bool] => bool" (‹(3Π_∈_./ _)› 10)
translations
  "Π b∈x. B" ⇌ "CONST depfun_ty (CONST MemOf x) (λb. B)"

context GZF begin

(*Lemmas taken from ZF/ZF_Base*)
lemma ballI [intro!]: "(⋀x. x ∈ A ⟹ P x) ⟹ ∀x∈A. P x"
  by (simp add: ball_def)

lemma bspec [dest?]: "[| ∀x∈A. P x;  x∈A |] ==> P x"
  unfolding ball_def by auto

lemma rev_ballE [elim]:
    "[| ∀x∈A. P x;  x∉A ==> Q;  P x ==> Q |] ==> Q"
  by (simp add: ball_def, blast)

lemma ballE: "[| ∀x∈A. P x;  P x ==> Q;  x∉A ⟹  Q |] ==> Q"
  by auto

lemma ball_triv [simp]: "∀x∈A. P ⟷ ((∃x. x∈A) ⟶ P)"
 by (simp add: ball_def)

lemma ball_cong [cong]:
 "[| A=A';  !!x. x∈A' ==> P x ⟷ P' x |] ==> ((∀x∈A. P x) ⟷ (∀x∈A'. P' x))"
  by (simp add: ball_def)

lemma bexI [intro]: "[| P x;  x ∈ A |] ==> ∃x∈A. P x"
  by (simp add: bex_def, blast)

lemma rev_bexI: "[| x∈A;  P x |] ==> ∃x∈A. P x" by blast

lemma bexCI: "[| ball A  (λx. ¬ P x) ⟹ P a;  a∈ A |] ==> ∃x∈A. P x"
  by blast

lemma bexE [elim!]: 
  "[|  ∃x∈A. P x ;  !!x. [| x∈A; P x |] ==> Q |] ==> Q"
by (simp add: bex_def, blast)

lemma bex_triv [simp]: "(∃x∈A. P) ⟷ ((∃x. x∈A) ∧ P)"
by (simp add: bex_def)

lemma bex_cong [cong]:
    "[| A=A';  !!x. x∈A' ==> P x ⟷ P'(x) |]
     ==> (∃x∈A. P x) ⟷ (∃x∈A'. P' x)"
  by (simp add: bex_def cong: conj_cong)

subsection ‹Rules for the subset relation›
(*Removed the need for x and y to be sets for x ⊆ y to be true.
  x ⊆ y now holds if ∀a. a ∈ x ⟶ a ∈ y.
  This change is in anticipation of having other objects that implement ∈
  (e.g., groups G ⊆ H being equivalent to carrier G ⊆ carrier H) *)

lemma subset_iff : 
  "x ⊆ y ⟷ (∀a ∈ x. a ∈ y)"
  using Subset_def by auto

lemma subsetI [intro]: 
  "(⋀a. a ∈ x ⟹ a ∈ y) ⟹ x ⊆ y"
  using subset_iff by auto
(*WAS: by (simp add: subset_def) *)

lemma subsetD [elim]: 
  "[| A ⊆ B;  c∈A |] ==> c∈B"
  using subset_iff by auto
(*WAS: apply (unfold subset_def)
       apply (erule bspec, assumption)
       done*)

lemma subsetCE [elim]: 
  "[| A ⊆ B;  c∉A ==> P;  c∈B ==> P |] ==> P"
  using subsetD by auto
(*WAS: by (simp add: subset_def, blast)*)

lemma rev_subsetD: 
  "[| c∈A; A ⊆ B |] ==> c∈B"
   by blast

lemma contra_subsetD: 
  "[| A ⊆ B; c ∉ B |] ==> c ∉ A"
   by blast

lemma rev_contra_subsetD: 
  "[| c ∉ B;  A ⊆ B |] ==> c ∉ A"
  by blast

lemma subset_refl : "A ⊆ A" 
  by blast

lemma subset_trans: 
   "[| A ⊆ B;  B ⊆ C |] ==> A ⊆ C"
  using subset_iff by blast
(*WAS: by blast*)
