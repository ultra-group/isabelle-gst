Theory ConnectionBase

theory ConnectionBase
  imports ModelTypes
begin

locale ConnectionBase = opening lifting_syntax +
  fixes Abs :: ‹('a :: ModelBase) ⇒ 'b›
    and Rep :: ‹'b ⇒ 'a›
    and rel :: ‹'a ⇒ 'b ⇒ bool› (infix ‹≃› 50)
  assumes tydef : ‹type_definition Rep Abs (Set.Collect (λx. x : M))›
      and rel_def : ‹rel x y = (x = Rep y)›
begin

interpretation c : type_definition ‹Rep :: 'b ⇒ 'a› ‹Abs :: 'a ⇒ 'b› ‹(Set.Collect (λx. x : M))›
  by (rule tydef)

lemma M_abs_inverse :
  assumes b:"b : M"
  shows "Rep (Abs b) = b"
  using c.Abs_inverse b by auto

lemma M_rep : 
  "Rep b : M"
  using c.Rep by auto

lemmas isom [simp] =  M_abs_inverse M_rep c.Rep_inverse 


lemma ceqD1 [dest] : "x ≃ y ⟹ Abs x = y"
  unfolding rel_def by auto

lemma ceqD2 [dest] : "x ≃ y ⟹ x = Rep y"
  unfolding rel_def by auto

lemma ceq_rep_refl [simp] : "(Rep x) ≃ x"
  unfolding rel_def by auto

lemma ceq_abs_refl [simp] : 
  assumes "x : M"
    shows "x ≃ (Abs x)"
  unfolding rel_def 
  using assms by auto

subsection ‹Transfer rules for logical constants›

lemma typing_transfer [transfer_rule]: "(R ===> (R ===> iff) ===> iff) (:) (:)"
  unfolding rel_fun_def has_ty_def by auto

lemma eq_transfer [transfer_rule] : 
  "((≃) ===> (≃) ===> (⟷)) (=) (=)" 
  by auto

lemma all_transfer [transfer_rule] :
  "(((≃) ===> (⟷)) ===> (⟷)) (mall) (All)"
proof (rule)
  fix P Q assume "((≃) ===> iff) P Q"
  hence x:"⋀x x'. x ≃ x' ⟹ P x ⟷ Q x'" by (auto dest: rel_funD)
  show "mall P ⟷ All Q"
    unfolding mall_def 
    using x[OF ceq_rep_refl] x[OF ceq_abs_refl] M_rep 
    by blast
qed

lemma ex_transfer [transfer_rule] :
  "(((≃) ===> (⟷)) ===> (⟷)) (mex) (Ex)"
proof (rule)
  fix P Q assume "((≃) ===> iff) P Q"
  hence x:"⋀x x'. x ≃ x' ⟹ P x ⟷ Q x'" by (auto dest: rel_funD)
  show "mex P ⟷ Ex Q"
    unfolding mex_def
    using x[OF ceq_rep_refl] x[OF ceq_abs_refl] M_rep

qed

lemma ex1_transfer [transfer_rule] :
  "(((≃) ===> (⟷)) ===> (⟷)) (mex1) (Ex1)"
proof (rule)
  fix P Q assume "((≃) ===> iff) P Q"
  hence x:"⋀x x'. x ≃ x' ⟹ P x ⟷ Q x'" by (auto dest: rel_funD)
  show "mex1 P ⟷ Ex1 Q"
  proof 
    assume P:"mex1 P" show "Ex1 Q"
    proof (insert P, unfold mex1_def tex1_def, erule ex1E, rule ex1I)
      fix x assume 
        "x : M ∧ P x" and uniq:"∀y. y : M ∧ P y ⟶ y = x"
      thus "Q (Abs x)" "⋀y. Q y ⟹ y = (Abs x)" 
      proof (metis x[OF ceq_abs_refl])
        fix y assume "Q y"
        hence "(Rep y) = x" using uniq x[OF ceq_rep_refl] by auto
        thus "y = Abs x" using c.Rep_inverse by auto 
      qed
    qed
  next
    assume Q:"Ex1 Q" show "mex1 P"
      unfolding mex1_def tex1_def
    proof (rule ex1E[OF Q], rule ex1I)
      fix y assume "Q y" and uniq:"∀z. Q z ⟶ z = y"
      thus "Rep y : M ∧ P (Rep y)" using M_rep[of y] x[OF ceq_rep_refl, of y] by metis
      fix y' assume y': "y' : M ∧ P y'" 
      hence "Abs y' = y" using uniq x[OF ceq_abs_refl, of y'] by auto 
      thus "y' = Rep y" using M_abs_inverse y' by auto 
    qed
  qed
qed

lemma uniq_transfer [transfer_rule] :
  "(((≃) ===> (⟷)) ===> (⟷)) (muniq) (Uniq)"
proof (rule)
  fix P Q assume "((≃) ===> iff) P Q"
  hence x:"⋀x x'. x ≃ x' ⟹ P x ⟷ Q x'" by (auto dest: rel_funD)
  show "muniq P ⟷ Uniq Q"
  proof 
    assume P:"muniq P" show "Uniq Q"
    proof (rule)
      fix x y assume "Q x" "Q y"
      hence "P (Rep x)" "P (Rep y)" using x[OF ceq_rep_refl] by auto
      hence "Rep x = Rep y" using P 
        unfolding muniq_def tuniq_def Uniq_def by auto
      thus "x = y" using c.Rep_inject by auto
     qed
  next
    assume Q:"Uniq Q" show "muniq P"
      unfolding muniq_def tuniq_def
    proof (rule, auto)
      fix x y assume "x : M" "P x" "y : M" "P y"
      hence "Q (Abs x)" "Q (Abs y)" using x[OF ceq_abs_refl] by auto
      hence "Abs x = Abs y" using Q unfolding Uniq_def by auto
      thus "x = y" using c.Abs_inject ‹x : M› ‹y : M› by auto
    qed
  qed
qed


lemma unary_pred_all_transfer [transfer_rule] :
  "((((≃) ===> iff) ===> iff) ===> iff) All All" 
proof 
  fix U :: "('a ⇒ bool) ⇒ bool" and V  
  assume "(((≃) ===> (⟷)) ===> (⟷)) U V"
  hence pq:"⋀P Q. (⋀a a' b b'. a ≃ a' ⟹ P a ⟷ Q a') ⟹ U P ⟷ V Q"
    unfolding rel_fun_def by auto
  show "All U = All V"
  proof
    assume U:"All U"
    show "All V"
    proof 
      fix Q :: "'b ⇒ bool"
      let ?P = "λa. Q (Abs a)"
      have "⋀a a'. a ≃ a' ⟹ ?P a ⟷ Q a'" by auto
      hence "U ?P ⟷ V Q" by (rule pq)
      thus "V Q" using U by simp
    qed
  next
    assume V: "All V"
    show "All U"  
    proof 
      fix P :: "'a ⇒ bool"
      let ?Q = "λa'. P (Rep a')"
      have "⋀a a'. a ≃ a' ⟹ P a ⟷ ?Q a'" by auto
      hence "U P ⟷ V ?Q" by (rule pq)
      thus "U P" using V by simp
    qed
  qed
qed

lemma binary_pred_all_transfer [transfer_rule] :
  "((((≃) ===> (≃) ===> iff) ===> iff) ===> iff) All All" 
proof 
  fix U :: "('a ⇒ 'a ⇒ bool) ⇒ bool" and V  
  assume "(((≃) ===> (≃) ===> (⟷)) ===> (⟷)) U V"
  hence pq:"⋀P Q. (⋀a a' b b'. a ≃ a' ⟹ b ≃ b' ⟹ P a b ⟷ Q a' b') ⟹ U P ⟷ V Q"
    unfolding rel_fun_def by auto
  show "All U = All V"
  proof
    assume U:"All U"
    show "All V"
    proof 
      fix Q :: "'b ⇒ 'b ⇒ bool"
      let ?P = "λa b. Q (Abs a) (Abs b)"
      have "⋀a a' b b'. a ≃ a' ⟹ b ≃ b' ⟹ ?P a b ⟷ Q a' b'" by auto
      hence "U ?P ⟷ V Q" by (rule pq)
      thus "V Q" using U by simp
    qed
  next
    assume V: "All V"
    show "All U"  
    proof 
      fix P :: "'a ⇒ 'a ⇒ bool"
      let ?Q = "λa' b'. P (Rep a') (Rep b')"
      have "⋀a a' b b'. a ≃ a' ⟹ b ≃ b' ⟹ P a b ⟷ ?Q a' b'" by auto
      hence "U P ⟷ V ?Q" by (rule pq)
      thus "U P" using V by simp
    qed
  qed
qed

lemma binary_fun_all_transfer [transfer_rule] : 
  "((((≃) ===> (≃) ===> (≃)) ===> (⟷)) ===> (⟷)) 
      (λP. ∀F : M → M → M. P F) (λP. ∀F. P F)"
proof
  fix P Q assume "(((≃) ===> (≃) ===> (≃)) ===> (⟷)) P Q"
  hence FG: "⋀F G. (⋀x x' y y'. x ≃ x' ⟹ y ≃ y' ⟹ F x y = Rep (G x' y')) ⟹ P F ⟷ Q G"
    unfolding rel_fun_def by auto
  show "(∀F : M → M → M. P F) ⟷ (∀F. Q F)"
  proof (rule)
    assume P:"∀F : M → M → M. P F"
    show "∀G. Q G"
    proof
      fix G
      let ?F = "λx y. Rep (G (Abs x) (Abs y))"
      have F_typ:"?F : M → M → M" by (rule funI, rule funI, auto)
      have "⋀x x' y y'. x ≃ x' ⟹ y ≃ y' ⟹ ?F x y = Rep (G x' y')" by (auto)
      hence "P ?F ⟷ Q G" using FG by auto
      thus "Q G" using P F_typ by auto
    qed
  next
    assume Q:"All Q"
    show "∀F : M → M → M. P F" 
    proof (rule) 
      fix F :: "'a ⇒ 'a ⇒ 'a" 
      assume F_typ:"F : M → M → M"
      let ?G = "λx y. Abs (F (Rep x) (Rep y))"
      have "⋀x x' y y'. x ≃ x' ⟹ y ≃ y' ⟹ F x y = Rep (?G x' y')" 
        using funE[OF funE[OF F_typ]] by auto
      hence "P F ⟷ Q ?G" using FG by auto
      thus "P F" using Q by auto
    qed
  qed
qed

subsection ‹Transfer rules for definite descriptions›

lemma defdes_transfer [transfer_rule] :
  "(rel ===> (rel ===> (⟷)) ===> rel) (λd P. mdefdes P d) the_default"
proof (rule, rule)
  fix d d' P P'
  assume d:"d ≃ d'" and "(rel ===> iff) P P'"
  hence P_iff :"⋀b b'. b ≃ b' ⟹ P b ⟷ P' b'" by (auto dest: rel_funD)
  show "(mdefdes P d) ≃ (ι' d' P')"
    unfolding rel_def mdefdes_def typ_defdes_def
  proof (cases ‹∃!x. P' x›, rule the_def_eq')
    case ex1:True
    hence "P' (ι' d' P')"
      using the_defI' by simp
    moreover have rel:"(Rep ι' d' P') ≃ ι' d' P'" by auto
    ultimately show "Rep ι' d' P' : M ∧ P (Rep ι' d' P')" 
      using P_iff[OF rel] M_rep by auto
  
    from ex1 obtain b' where "P' b'" and c':"⋀c'. P' c' ⟹ c' = b'" by auto
    hence "P (Rep b')" "⋀c. c : M ∧ P c ⟹ c = Rep b'" 
    proof (metis P_iff[of ‹Rep b'› ‹b'›, OF ceq_rep_refl])
      fix c assume c:"c : M ∧ P c"
      hence "P' (Abs c)" 
        using P_iff[OF ceq_abs_refl] by auto
      hence "Abs c = b'" using c' by simp
      thus "c = Rep b'" using c by auto
    qed 
    thus "∃!x'. x' : M ∧ P x'" 
      using ex1I[of ‹λx'. x' : M ∧ P x'› ‹Rep b'›] by auto
  next
    case nex1P: False
    hence nex1P':"¬ (∃!x. x : M ∧ P x)" using P_iff 
      by (metis M_rep c.Rep_inverse ceq_abs_refl) 
    have "(ι x. x : M ∧ P x else d) = d"
     and "ι' d' P' = d'" 
    using the_def_default[OF nex1P] the_def_default[OF nex1P'] by auto
    then show "(ι x. x : M ∧ P x else d) = Rep ι' d' P'"
      using d by auto
  qed
qed
