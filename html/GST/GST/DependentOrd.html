Theory DependentOrd

theory DependentOrd
  imports Ordinal "../OPair/DependentSum" 
begin

abbreviation (in Ordinal) (input) gt where "gt j i ≡ lt i j"

syntax
  "_depfun_ord" :: "[pttrn, 'a, bool] => bool" (‹(3Π_<_./ _)› 10)
  "_depsum_ord" :: "[pttrn, 'a, bool] => bool" (‹(3Σ_<_./ _)› 10)
translations
  "Π i<j. B" ⇌ "Π i:(CONST Ord ¦ (CONST gt) j). B"
  "Σ i<j. B" ⇌ "Σ i:(CONST Ord ¦ (CONST gt) j). B"

context Ordinal begin

subsection ‹Rules for dependent functions on ordinals›

lemma depfun_ordI :
  assumes i:"⋀i. i : Ord ⟹ i < j ⟹ F i :  B i"
  shows "F : (Π i < j. B i)"
  by (rule depfunI, drule intE, use i in unfold_typs)

lemma depfun_ordE :
  assumes F:"F : (Π i < j. B i)" 
      and i:"i : Ord" "i < j"
  shows "F i : B i"
  by (rule depfunE[OF F intI[OF i(1) tyI]], rule i(2))

end

class DepSumOrd = 
  OPair + Ordinal

begin

subsection ‹Rules for dependent sums on ordinals›

lemma depsum_ordI : 
  assumes ib:"<i,b> : Pair" 
      and i:"i : Ord" "i < j" and b:"b : B i" 
    shows "<i,b> : (Σ i < j. B i)"
  by (rule depsumI[OF ib intI[OF i(1) tyI]], rule i(2), rule b)

lemma depsum_ordD : 
  assumes t:"t : (Σ i < j. B i)"
  shows "t : Pair" "τ t : Ord" "τ t < j" "π t : B (τ t)"
  by (use depsumD[OF t] in unfold_typs)

lemma depsum_ordD_pair : 
  assumes t:"<i,b> : (Σ i < j. B i)"
  shows "<i,b> : Pair" "i : Ord" "i < j" "b : B i"
  by (use depsumD_pair[OF t] in unfold_typs)

lemma depsum_ordE : 
  assumes t:"t : (Σ i < j. B i)"
  obtains i b where "t : Pair" "t = <i,b>" "i : Ord" "i < j" "b : B i"
  by (rule depsumE[OF t], unfold_typs)

end
end
