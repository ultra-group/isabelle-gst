Theory DependentSum

theory DependentSum
  imports OPair
begin

context OPair begin

subsection ‹Dependent Sum Type›

definition depsum_ty :: "('a ⇒ bool) ⇒ ('a ⇒ ('a ⇒ bool)) ⇒ ('a ⇒ bool)" 
  where "depsum_ty α β ≡ Pair △ (λp. τ p : α ∧ π p : β (τ p))"

end

syntax
  "_depsum_ty" :: "[pttrn, 'a ⇒ bool, bool] => bool" (‹(3Σ_:_./ _)› 10)
(*What does CONST do?*)
translations
  "Σ x:A. B" ⇌ "CONST depsum_ty A (λx. B)"

context OPair begin
lemma depsumI : 
  assumes ab:"<a,b> : Pair" 
     and a:"a : A" and b:"b : B a" 
  shows "<a,b> : (Σ a : A. B a)"
  unfolding depsum_ty_def
  by (rule intI[OF ab], 
      rule tyI, unfold fst_eq[OF ab] snd_eq[OF ab],
      simp only: a b)

lemma depsumI_eq : 
  assumes p:"p : Pair" "p = <a,b>" 
      and a:"a : A" and b:"b : B a" 
    shows "p : (Σ a : A. B a)"
  using depsumI[of a b A B, OF _ a b] p by auto

lemma depsumD : 
  assumes t:"t : (Σ a : A. B a)"
  shows "t : Pair" "τ t : A" "π t : B (τ t)"
  by (insert t, unfold depsum_ty_def, unfold_typs)

lemma depsumD_pair : 
  assumes t:"<a,b> : (Σ a : A. B a)"
  shows "<a,b> : Pair" "a : A" "b : B a"
  using depsumD[OF t] fst_eq snd_eq by auto

lemma depsumE : 
  assumes t:"t : (Σ a : A. B a)"
  obtains a b where "t : Pair" "t = <a,b>" "a : A" "b : B a" 
  using depsumD[OF t] pair_proj_eq by blast

lemma depsum_pair_subtyp : 
  "(Σ a : A. B a) << Pair"
  by (rule subtypI, drule depsumD(1), assumption)
subsection ‹Product soft-type›

definition Product (infix ‹*› 50) 
  where [typdef] : "α * β ≡ Σ x : α. β"

lemma productI [typ_intro] : 
  assumes "<a,b> : Pair" "a : α" "b : β" 
  shows "<a,b> : α * β"
  unfolding Product_def 
  by (rule depsumI, use assms in auto)

lemma productD : 
  assumes t:"t : α * β"
  shows "t : Pair" "τ t : α" "π t : β"
  by (insert t, unfold Product_def, use depsumD in auto)

lemma productD_pair : 
  assumes t:"<a,b> : α * β"
  shows "<a,b> : Pair" "a : α" "b : β"
proof -
  from t have ab:"<a,b> : (Σ a:α. β)" by (unfold Product_def)
  show "<a,b> : Pair" "a : α" "b : β" using depsumD_pair[OF ab] by auto
qed

lemma productE :
  assumes "p : α * β"
  obtains a b where "p : Pair" "p = <a,b>" "a : α" "b : β"
  using assms unfolding Product_def 
  by (blast elim: depsumE)

lemma prod_pair : "p : α * β ⟹ p : Pair" 
  by (rule productE)

end
end
