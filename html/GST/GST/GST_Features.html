    SetMem_ax   : "∀b. SetMem b = (∃y : Set. b ∈ y)" and
    SetOf_ax    : "∀α x. SetOf α x = (x : Set ∧ (∀b. b ∈ x ⟶ b : α))" and
    ReplPred_ax : "∀x P. ReplPred x P = (∀a. a ∈ x ⟶ (∃⇩≤⇩1 b: SetMem. P a b))"

  fixes 
    ― ‹Axiomatized constants›
    BinRel :: ‹'a ⇒ bool› and 
    mkrel :: ‹['a, 'a, ['a,'a] ⇒ bool] ⇒ 'a› and

    ― ‹Derived constants›
    BinRelMem :: ‹'a ⇒ bool› and
    BinRelPred :: ‹['a, 'a, ['a,'a] ⇒ bool] ⇒ bool› and
    ― ‹Default value›
    BinRelation_default :: ‹'a›
  assumes 
    ― ‹Soft typings›
    mkrel_typ : "mkrel : (Π x : Set. Π y : Set. BinRelPred x y → BinRel)" and 
    field_typ : "field : BinRel → Set" and
    ― ‹Main axioms›
    rel : "∀x : Set. ∀y : Set. ∀P : BinRelPred x y. 
              (∀a b. app (mkrel x y P) a b ⟷ (a ∈ x ∧ b ∈ y ∧ P a b))" and
    ext : "∀r : BinRel. ∀s : BinRel. (∀a b. app r a b ⟷ app s a b) ⟶ r = s" and
    field_iff : "∀r : BinRel. ∀x. x ∈ field r ⟷ (∃y. app r x y ∨ app r y x)" and
    ― ‹Simple definitions›
    BinRelMem_def : "BinRelMem = app_mem BinRel" and
    BinRelPred_def : "BinRelPred = 
      (λx y P.  ∀a b. a ∈ x ⟶ b ∈ y ⟶ P a b ⟶ a : BinRelMem ∧ b : BinRelMem)"

class Ordinal =
  fixes 
    ― ‹Axiomatized constants›
    Ord :: ‹'a ⇒ bool› and 
    lt  :: ‹['a, 'a] ⇒ bool› (infix ‹<› 50) and
    zero  :: ‹'a› (‹0›) and
    succ  :: ‹'a ⇒ 'a› and
    omega :: ‹'a› (‹ω›) and
    ― ‹Derived constants›
    Limit :: ‹'a ⇒ bool› and
    ― ‹Default value›
    Ordinal_default :: ‹'a›
  assumes 
    ― ‹Soft typings›
    zero_typ  : "0 : Ord" and 
    succ_typ  : "succ : Ord → Ord" and
    omega_typ : "ω : Limit" and
    ― ‹Main axioms›
    zero_ax  : "∀b : Ord. ¬ (b < 0)" and
    succ_ax  : "∀a : Ord. ∀b : Ord. a < (succ b) ⟷ a < b ∨ a = b" and
    omega_ax : "∀μ : Limit. μ = ω ∨ ω < μ" and
    lt_trans  : "∀i : Ord. ∀j : Ord. ∀k : Ord. i < j ⟶ j < k ⟶ i < k" and
    lt_notsym : "∀i : Ord. ∀j : Ord. i < j ⟶ ¬ (j < i)" and
    lt_linear : "∀i : Ord. ∀j : Ord. i < j ∨ i = j ∨ j < i" and
    lt_induct : "∀P. (∀i : Ord. (∀j : Ord. j < i ⟶ P j) ⟶ P i) ⟶ (∀a : Ord. P a)" and
    ― ‹Simple definitions›
    Limit_def : "Limit = (Ord △ (λμ. 0 < μ ∧ (∀j : Ord. j < μ ⟶ succ j < μ)))"
