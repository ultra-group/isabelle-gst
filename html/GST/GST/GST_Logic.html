(*Perhaps not use 'a as the type variable *)
definition the_default :: "'a ⇒ ('a ⇒ bool) ⇒ 'a" (‹ι'' _ _› [90,90] ) where
  "ι' d F ≡ if Ex1 F then (THE x. F x) else d"

syntax 
  "_the_default" :: "[idts, 'a ⇒ bool, 'a] ⇒ bool" (‹(3ι_./ _/ else _)›  [0, 10, 10] 10)
translations
  "ι x. P else default" ⇌ "CONST the_default default (λx. P)"

lemma the_def_if : "if (Ex1 F) then (F a ⟷ ι' d F = a) 
                               else ι' d F = d"
proof (cases ‹Ex1 F›)
  assume "Ex1 F" 
  hence eq:"ι' d F = (THE x. F x)" unfolding the_default_def by auto
  have "F a ⟷ ι' d F = a" 
  proof
    assume "F a"
    thus "ι' d F = a" by (subst eq, insert ‹Ex1 F›, auto)
  next
    assume "ι' d F = a"
    thus "F a" using eq ‹Ex1 F› by (auto intro: theI)
  qed
  thus "if (Ex1 F) then (F a ⟷ ι' d F = a) else ι' d F = d" using ‹Ex1 F› by simp
next
  assume "¬ Ex1 F"
  thus "if (Ex1 F) then (F a ⟷ ι' d F = a) else ι' d F = d" 
    unfolding the_default_def by auto
qed

lemma the_def_default :
  assumes "¬ (Ex1 P)" 
  shows "ι' d P = d"
  using the_def_if[of P] assms by auto

lemma the_def_eq : 
  assumes "P a" "⋀x. P x ⟹ x = a"
  shows "ι' d P = a"
proof -
  from assms have "Ex1 P" by auto
  hence "P a ⟷ (ι' d P) = a" using the_def_if[of P] by auto
  thus "(ι' d P) = a" using ‹P a› by auto
qed

lemma the_def_eq' : "⟦ Ex1 P ; P a ⟧ ⟹ (ι' d P) = a"
  by (blast intro: the_def_eq)

lemma the_defI : assumes "P a" "⋀x. P x ⟹ x = a"
  shows "P (ι' d P)"
proof -
  from the_def_eq[of P] assms have "ι' d P = a" by blast 
  thus "P (ι' d P)" using ‹P a› by simp
qed

(*Definite description lemmas followed by an apostrophe use an 
 explicit uniqueness quantifier in their assumptions*)
lemma the_defI' : assumes "Ex1 P" shows "P (ι' d P)"
  using assms by (blast intro: the_defI)

(*Distinction of _ and 2 is from Isabelle/HOL's lemma names *)
lemma the_defI2 : assumes "P a" "⋀x. P x ⟹ x = a" "⋀x. P x ⟹ Q x"
  shows "Q (ι' d P)" by (blast intro: assms the_defI)

lemma the_defI2' : assumes "Ex1 P" "⋀x. P x ⟹ Q x" shows "Q (ι' d P)" 
  by (blast intro: assms(2) the_defI2[where P=P and Q=Q] ex1E[OF assms(1)])

lemma the_def_cong : 
  assumes "P = Q"
  shows "(ι' d P) = (ι' d Q)"
  unfolding assms ..

declare the_defI2' [simp]  

subsection ‹Restricted quantification using a predicate›
(*TODO: remove these, and just define ball, bex in terms of tall, tex.*)

definition rall :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool" where
  rall_def : "rall P Q ≡ ∀x. P x ⟶ Q x"
definition rex :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool" where
  rex_def : "rex P Q ≡ ∃x. P x ∧ Q x"
definition rex_uniq :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool" where
  rex_uniq_def : "rex_uniq P Q ≡ (∃⇩≤⇩1 x. P x ∧ Q x)"
definition rex1 :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool" where
  rex1_def : "rex1 P Q ≡ (∃! x. P x ∧ Q x)"

syntax
  "_rall"     :: "['a=>bool, pttrn, bool] => bool" (‹(3∀[_]_./ _)› 10)
  "_rex"      :: "['a=>bool, pttrn,  bool] => bool" (‹(3∃[_]_./ _)› 10)
  "_rex_uniq" :: "['a=>bool, pttrn, bool] => bool" (‹(3∃⇩≤⇩1[_]_./ _)› 10)
  "_rex1"     :: "['a=>bool, pttrn, bool] => bool" (‹(3∃![_]_./ _)› 10)
translations
  "∀[M]x. P" ⇌ "CONST rall M (λx. P)"
  "∃[M]x. P" ⇌ "CONST rex M  (λx. P)"
  "∃⇩≤⇩1[M] x. P" ⇌ "CONST rex_uniq M  (λx. P)"
  "∃![M] x. P" ⇌ "CONST rex1 M  (λx. P)"

(*Taken from ZF/OrdQuant, unfolded syntax*)
(*Go back and take from OrdQuant*)
lemma rallI [intro]: "[| !!x. M(x) ==> P(x) |] ==> ∀[M]x. P x"
by (simp add: rall_def)

lemma rspec: "[| rall M P; M(x) |] ==> P(x)"
by (simp add: rall_def)

lemma rev_rallE [elim]:
    "[| rall M P;  ¬ M(x) ==> Q;  P(x) ==> Q |] ==> Q"
by (simp add: rall_def, blast)

lemma rallE: "[| rall M P; P(x) ==> Q;  ¬ M(x) ==> Q |] ==> Q"
by blast

lemma rall_triv [simp]: "(rall M (λx. P)) ⟷ ((∃x. M x) ⟶ P)"
by (simp add: rall_def)

lemma rall_cong [cong]:
    "(⋀x. M(x) ⟹ P(x) ⟷ P'(x)) ==> (rall M P) ⟷ (rall M P')"
by (simp add: rall_def)

lemma rexI [intro]: "[| P(x); M(x) |] ==> rex M P"
by (simp add: rex_def, blast)

lemma rev_rexI: "[| M(x);  P(x) |] ==> rex M P"
by blast

lemma rexCI: "[| rall M (λx. ¬P x) ==> P(a); M(a) |] ==> rex M P"
by blast

lemma rexE [elim]: "[| rex M P;  !!x. [| M(x); P(x) |] ==> Q |] ==> Q"
by (simp add: rex_def, blast)

lemma rex_triv [simp]: "(rex M (λx. P)) ⟷ ((∃x. M(x)) ∧ P)"
by (simp add: rex_def)

lemma rex_cong [cong]:
    "(!!x. M(x) ==> P(x) ⟷ P'(x)) ==> (rex M P) ⟷ (rex M P')"
by (simp add: rex_def cong: conj_cong)

lemma atomize_rall: "(!!x. M(x) ==> P(x)) == Trueprop (rall M P)"
by (simp add: rall_def atomize_all atomize_imp)
