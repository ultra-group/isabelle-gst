lemma setof_subset : 
  assumes "x : Set" "y : SetOf α" "x ⊆ y"
    shows "x : SetOf α"
proof (rule setofI[OF ‹x : Set›])
  fix b assume "b ∈ x"
  hence "b ∈ y" by (rule subsetD[OF ‹x ⊆ y›])
  thus "b : α" by (rule setof_mem[OF ‹y : SetOf α›])
qed

subsection ‹Equality of sets›
lemma equality_iff : 
  assumes "x : Set" "y : Set"
    shows "x = y ⟷ (∀b. b ∈ x ⟷ b ∈ y)"
  using assms ext 
  by auto

lemma subset_equal : 
  assumes "x : Set" "y : Set" 
    shows "x = y ⟷ x ⊆ y ∧ y ⊆ x"
  using equality_iff[OF assms] subset_iff assms 
  by auto

(*Lemmas taken from ZF/ZF_Base:*)
lemma equalityI : 
  assumes "x : Set" "y : Set" 
    shows "⟦ x ⊆ y;  y ⊆ x ⟧ ⟹ x = y"
  using subset_equal[OF assms] 
  by auto
(*WAS: by (rule extension [THEN iffD2], rule conjI) *)

(*Added assumptions: ‹A : Set› ‹B : Set›*)
lemma equality_iffI: 
  assumes "x : Set" "y : Set" 
    shows "(⋀b. b ∈ x ⟷ b ∈ y) ⟹ x = y"
  using assms equalityI subsetI 
  by blast
(*WAS: by (rule equalityI, blast+)*)

(*Replaced ‹extension› with ‹subset_equal›*)

    shows "[| A = B;  [| A ⊆ B; B ⊆ A |] ==> P |]  ==>  P"

subsection ‹Rules for power sets›

(*Added assumption: ‹x : Set›*)
lemma pow_iff : 
  assumes "x : Set" "y : Set" 
    shows "x ∈ 𝒫 y ⟷ x ⊆ y"
  using pow assms by auto
(*WAS: an axiom*)

lemma Pow_typ_set : "Pow : Set → Set"
  by (rule funI[OF subtypE[OF setof_set_subtyp funE[OF Pow_typ]]])

(*Soft typing rules can be derived from instances of typing rules: *)
lemmas pow_setof = funE[OF Pow_typ]
lemmas pow_set   = funE[OF Pow_typ_set]

lemma pow_mem : "x : Set ⟹ y ∈ 𝒫 x ⟹ y : Set"
  using funE[OF Pow_typ, THEN tyE] unfolding SetOf_def by auto

(*Added assumption: ‹x : Set›*)
lemma powI: 
  assumes "x : Set" "y : Set" 
    shows "x ⊆ y ⟹ x ∈ 𝒫 y"
  using pow_iff[OF assms] by auto
(*WAS: by (erule Pow_iff [THEN iffD2])*)

lemma powD: 
  assumes y : "y : Set"  
    shows "x ∈ 𝒫 y ⟹ x ⊆ y"
  using pow_iff[OF setof_mem[OF pow_setof[OF y]] assms] by simp
(*WAS: by (erule Pow_iff [THEN iffD1]) *)

lemma pow_top : 
  assumes x : "x : Set" 
    shows "x ∈ 𝒫 x"
  by (rule powI[OF x x subset_refl])

lemma set_setmem_subtyp : "Set << SetMem"
proof (rule subtypI, rule setmemI)
  fix x assume "x : Set"
  thus "𝒫 x : Set" "x ∈ 𝒫 x" by (rule pow_set, rule pow_top)
qed

lemma pow_iterate :
  assumes x : "x : Set" and y : "y : Set" 
      and pow : "x ∈ 𝒫 y"
    shows "𝒫 x ∈ 𝒫 (𝒫 y)"
proof (rule powI[OF pow_set[OF x] pow_set[OF y]], rule)
  fix a assume "a ∈ 𝒫 x" 
  hence "a : Set" "a ⊆ x" 
    using setof_mem[OF pow_setof[OF x]] powD[OF x] by auto
  hence "a ⊆ y" using subset_trans[OF _ powD[OF y pow]] by auto
  thus "a ∈ 𝒫 y" using powI[OF ‹a : Set› y] by auto
qed

subsection ‹Rules for Replacement and derived operators›

thm ReplPred_def
lemma replpredI :
  assumes "x : Set" "⋀a. a ∈ x ⟹ ∃⇩≤⇩1 b : SetMem. P a b"
  shows "P : ReplPred x"
  using assms
  unfolding ReplPred_def has_ty_def by auto

lemma repl_iff : 
  assumes "x : Set" "P : ReplPred x"
    shows "∀b. b ∈ { b | ∃a∈x. P a b } ⟷ (∃a. a ∈ x ∧ P a b) ∧ b : SetMem"
  using repl assms by blast

lemma repl_set : 
  assumes "x : Set" "P : ReplPred x"
    shows "{ b | ∃a∈x. P a b } : Set"
  using funE[OF depfunE[OF Repl_typ]] assms 
  by auto

lemma replace_iff : 
  assumes "x : Set" "P : ReplPred x" 
    shows "b ∈ { b | ∃a∈x. P a b } ⟷ (∃a ∈ x. P a b) ∧ b : SetMem"
  using repl assms unfolding bex_def rex_def 
  by blast

lemma replaceI : 
  assumes "x : Set" "P : ReplPred x" 
    shows "⟦ P a b ; a ∈ x ; b : SetMem⟧ ⟹ b ∈ { b | ∃a∈x. P a b }"
  unfolding replace_iff[OF assms] 
  by blast   

lemma replaceE: 
  assumes "x : Set" "P : ReplPred x" 
    shows "⟦ b ∈ { b | ∃a∈x. P a b } ; ⋀a. ⟦ a ∈ x ; b : SetMem ; P a b ⟧ ⟹ R ⟧ ⟹ R"
  using replace_iff[OF assms] by blast
  
lemma replace_cong : 
  assumes "x : Set" "P : ReplPred x" "y : Set" "Q : ReplPred y"  
    shows [cong]: "⟦ x = y ; ⋀a b. a ∈ y ⟹ P a b ⟷ Q a b ⟧ ⟹ Repl x P = Repl y Q"
  by (rule equality_iffI[OF repl_set[OF assms(1,2)] repl_set[OF assms(3,4)]],
      unfold replace_iff[OF assms(1,2)] replace_iff[OF assms(3,4)], blast)

definition ReplFunPred :: "['a, 'a ⇒ bool] ⇒ ((['a,'a] ⇒ bool) ⇒ bool)"
  where [typdef] : "ReplFunPred x β ≡ ReplPred x △ BinPred (MemOf x) β"

lemma funpred_replpred: "ReplFunPred x β << ReplPred x"  
  unfolding ReplFunPred_def 
  by (rule int_sub1)

lemma funpred_binpred_typ : "ReplFunPred x β << BinPred (MemOf x) β"
  unfolding ReplFunPred_def 
  by (rule int_sub2)

lemmas funpred_binpred = subtypE[OF funpred_binpred_typ]

lemma Repl_typ2 : "Repl : (Π x : Set. ReplFunPred x α → SetOf α)"
proof (rule depfunI, rule funI)  
  fix x P assume P:"P : ReplFunPred x α" and x:"x : Set" 
  hence "P : ReplPred x" 
    using subtypE[OF funpred_replpred]  by simp_all
  show "Repl x P : SetOf α"
  proof (rule setofI, rule repl_set[OF ‹x : Set› ‹P : ReplPred x›])
    fix b assume "b ∈ Repl x P" thus "b : α" 
    proof (rule replaceE[OF ‹x : Set› ‹P : ReplPred x›])
      fix a assume "a ∈ x" "P a b"
      thus "b : α"
        using bpredE[OF funpred_binpred[OF P] memofI] 
      by auto
    qed
  qed
qed
