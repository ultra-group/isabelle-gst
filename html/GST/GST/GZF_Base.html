lemma setof_subset : 
  assumes "x : Set" "y : SetOf Œ±" "x ‚äÜ y"
    shows "x : SetOf Œ±"
proof (rule setofI[OF ‚Äπx : Set‚Ä∫])
  fix b assume "b ‚àà x"
  hence "b ‚àà y" by (rule subsetD[OF ‚Äπx ‚äÜ y‚Ä∫])
  thus "b : Œ±" by (rule setof_mem[OF ‚Äπy : SetOf Œ±‚Ä∫])
qed

subsection ‚ÄπEquality of sets‚Ä∫
lemma equality_iff : 
  assumes "x : Set" "y : Set"
    shows "x = y ‚ü∑ (‚àÄb. b ‚àà x ‚ü∑ b ‚àà y)"
  using assms ext 
  by auto

lemma subset_equal : 
  assumes "x : Set" "y : Set" 
    shows "x = y ‚ü∑ x ‚äÜ y ‚àß y ‚äÜ x"
  using equality_iff[OF assms] subset_iff assms 
  by auto

(*Lemmas taken from ZF/ZF_Base:*)
lemma equalityI : 
  assumes "x : Set" "y : Set" 
    shows "‚ü¶ x ‚äÜ y;  y ‚äÜ x ‚üß ‚üπ x = y"
  using subset_equal[OF assms] 
  by auto
(*WAS: by (rule extension [THEN iffD2], rule conjI) *)

(*Added assumptions: ‚ÄπA : Set‚Ä∫ ‚ÄπB : Set‚Ä∫*)
lemma equality_iffI: 
  assumes "x : Set" "y : Set" 
    shows "(‚ãÄb. b ‚àà x ‚ü∑ b ‚àà y) ‚üπ x = y"
  using assms equalityI subsetI 
  by blast
(*WAS: by (rule equalityI, blast+)*)

(*Replaced ‚Äπextension‚Ä∫ with ‚Äπsubset_equal‚Ä∫*)

(*Added assumptions: ‚ÄπA : Set‚Ä∫ ‚ÄπB : Set‚Ä∫*)
lemma equalityE: 
  assumes "A : Set" "B : Set" 
    shows "[| A = B;  [| A ‚äÜ B; B ‚äÜ A |] ==> P |]  ==>  P"
  by (use equalityD1[OF assms] equalityD2[OF assms] in auto)
(*WAS: by (blast dest: equalityD1 equalityD2) *)

lemma pow_iff : 
  assumes "x : Set" "y : Set" 
    shows "x ‚àà ùí´ y ‚ü∑ x ‚äÜ y"
  using pow assms by auto
(*WAS: an axiom*)

lemma Pow_typ_set : "Pow : Set ‚Üí Set"
  by (rule funI[OF subtypE[OF setof_set_subtyp funE[OF Pow_typ]]])

(*Soft typing rules can be derived from instances of typing rules: *)
lemmas pow_setof = funE[OF Pow_typ]
lemmas pow_set   = funE[OF Pow_typ_set]

lemma pow_mem : "x : Set ‚üπ y ‚àà ùí´ x ‚üπ y : Set"
  using funE[OF Pow_typ, THEN tyE] unfolding SetOf_def by auto

(*Added assumption: ‚Äπx : Set‚Ä∫*)
lemma powI: 
  assumes "x : Set" "y : Set" 
    shows "x ‚äÜ y ‚üπ x ‚àà ùí´ y"
  using pow_iff[OF assms] by auto
(*WAS: by (erule Pow_iff [THEN iffD2])*)

lemma powD: 
  assumes y : "y : Set"  
    shows "x ‚àà ùí´ y ‚üπ x ‚äÜ y"
  using pow_iff[OF setof_mem[OF pow_setof[OF y]] assms] by simp
(*WAS: by (erule Pow_iff [THEN iffD1]) *)

lemma pow_top : 
  assumes x : "x : Set" 
    shows "x ‚àà ùí´ x"
  by (rule powI[OF x x subset_refl])

lemma set_setmem_subtyp : "Set << SetMem"
proof (rule subtypI, rule setmemI)
  fix x assume "x : Set"
  thus "ùí´ x : Set" "x ‚àà ùí´ x" by (rule pow_set, rule pow_top)
qed

lemma pow_iterate :
  assumes x : "x : Set" and y : "y : Set" 
      and pow : "x ‚àà ùí´ y"
    shows "ùí´ x ‚àà ùí´ (ùí´ y)"
proof (rule powI[OF pow_set[OF x] pow_set[OF y]], rule)
  fix a assume "a ‚àà ùí´ x" 
  hence "a : Set" "a ‚äÜ x" 
    using setof_mem[OF pow_setof[OF x]] powD[OF x] by auto
  hence "a ‚äÜ y" using subset_trans[OF _ powD[OF y pow]] by auto
  thus "a ‚àà ùí´ y" using powI[OF ‚Äπa : Set‚Ä∫ y] by auto
qed

lemma repl_iff :

    shows "‚àÄb. b ‚àà { b | ‚àÉa‚ààx. P a b } ‚ü∑ (‚àÉa. a ‚àà x ‚àß P a b) ‚àß b : SetMem"

  assumes "x : Set" "P : ReplPred x" 
    shows "b ‚àà { b | ‚àÉa‚ààx. P a b } ‚ü∑ (‚àÉa ‚àà x. P a b) ‚àß b : SetMem"

lemma replaceI : 
  assumes "x : Set" "P : ReplPred x" 
    shows "‚ü¶ P a b ; a ‚àà x ; b : SetMem‚üß ‚üπ b ‚àà { b | ‚àÉa‚ààx. P a b }"
  unfolding replace_iff[OF assms]

lemma replaceE: 
  assumes "x : Set" "P : ReplPred x" 
    shows "‚ü¶ b ‚àà { b | ‚àÉa‚ààx. P a b } ; ‚ãÄa. ‚ü¶ a ‚àà x ; b : SetMem ; P a b ‚üß ‚üπ R ‚üß ‚üπ R"

lemma replace_cong : 
  assumes "x : Set" "P : ReplPred x" "y : Set" "Q : ReplPred y"  
    shows [cong]: "‚ü¶ x = y ; ‚ãÄa b. a ‚àà y ‚üπ P a b ‚ü∑ Q a b ‚üß ‚üπ Repl x P = Repl y Q"
  by (rule equality_iffI[OF repl_set[OF assms(1,2)] repl_set[OF assms(3,4)]],
      unfold replace_iff[OF assms(1,2)] replace_iff[OF assms(3,4)], blast)

definition ReplFunPred :: "['a, 'a ‚áí bool] ‚áí ((['a,'a] ‚áí bool) ‚áí bool)"
  where [typdef] : "ReplFunPred x Œ≤ ‚â° ReplPred x ‚ñ≥ BinPred (MemOf x) Œ≤"

lemma funpred_replpred: "ReplFunPred x Œ≤ << ReplPred x"  
  unfolding ReplFunPred_def 
  by (rule int_sub1)

lemma funpred_binpred_typ : "ReplFunPred x Œ≤ << BinPred (MemOf x) Œ≤"
  unfolding ReplFunPred_def 
  by (rule int_sub2)

lemmas funpred_binpred = subtypE[OF funpred_binpred_typ]

lemma Repl_typ2 : "Repl : (Œ† x : Set. ReplFunPred x Œ± ‚Üí SetOf Œ±)"
proof (rule depfunI, rule funI)  
  fix x P assume P:"P : ReplFunPred x Œ±" and x:"x : Set" 
  hence "P : ReplPred x" 
    using subtypE[OF funpred_replpred]  by simp_all
  show "Repl x P : SetOf Œ±"
  proof (rule setofI, rule repl_set[OF ‚Äπx : Set‚Ä∫ ‚ÄπP : ReplPred x‚Ä∫])
    fix b assume "b ‚àà Repl x P" thus "b : Œ±" 
    proof (rule replaceE[OF ‚Äπx : Set‚Ä∫ ‚ÄπP : ReplPred x‚Ä∫])
      fix a assume "a ‚àà x" "P a b"
      thus "b : Œ±"
        using bpredE[OF funpred_binpred[OF P] memofI] 
      by auto
    qed
  qed
qed
