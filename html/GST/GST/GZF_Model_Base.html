definition mSet :: ‹'a ⇒ bool›
  where "mSet ≡ M △ (▸ set)"

definition mMem :: ‹'a ⇒ 'a ⇒ bool› (infix ‹m› 50) 
where "x m y ≡ y : mSet ∧ x ∈ snd y"

definition mSetMem :: ‹'a ⇒ bool›
  where "mSetMem b ≡ m∃x : mSet. b m x"

definition mSubset :: ‹['a, 'a] ⇒ bool›
  where "mSubset x y ≡ m∀a. a m x ⟶ a m y"

definition mSetOf :: ‹['a ⇒ bool, 'a] ⇒ bool›
  where "mSetOf P x ≡ x : mSet ∧ (m∀b. b m x ⟶ b : P)"

definition mReplPred :: ‹['a, ['a,'a] ⇒ bool] ⇒ bool›
  where "mReplPred x P ≡ m∀a. a m x ⟶ (m∃⇩≤⇩1 b : mSetMem. P a b)" 

definition mUnion' :: ‹'a ⇒ 'a› 
  where "mUnion' x ≡ <set, ⋃ y ∈ snd x. snd y>"

definition mUnion :: ‹'a ⇒ 'a› (‹m⋃›)
  where "m⋃ x ≡ if x : mSetOf mSet then mUnion' x
                                    else GZF_Model_mdefault"

definition mPow' :: ‹'a ⇒ 'a› 
  where "mPow' x ≡ <set, set ⊕ 𝒫 (snd x)>"

definition mPow :: ‹'a ⇒ 'a› (‹m𝒫›)
  where "m𝒫 x ≡ if x : mSet then mPow' x 
                            else GZF_Model_mdefault"

definition mEmp :: ‹'a› (‹m∅›)
  where "mEmp = <set, ∅>" 

definition mSucc' :: ‹'a ⇒ 'a›
  where "mSucc' x = mUnion' <set, {x, <set, {x}>}>"

definition mSucc :: ‹'a ⇒ 'a› 
  where "mSucc x ≡ if x : mSet then mSucc' x
                      else GZF_Model_mdefault"

definition ord_to_mvnord :: ‹'a ⇒ 'a› (‹Θ›)
  where ‹Θ ≡ OrdRec (λμ f. <set, {f j | j < μ}>) (λ_ x. mSucc x) m∅›

definition mInf :: ‹'a›
  where ‹mInf ≡ Θ ω›

definition mRepl' :: ‹['a, ['a,'a] ⇒ bool] ⇒ 'a› 
  where "mRepl' x P ≡ <set, { b | ∃ a ∈ snd x. P a b ∧ b : mSetMem }>" 

definition mRepl :: ‹['a, ['a,'a] ⇒ bool] ⇒ 'a› (‹mℛ›)
  where "mℛ x P ≡ if x : mSet ∧ P : mReplPred x 
                    then mRepl' x P else GZF_Model_mdefault"

theorem variants_set :
  shows "Variants set 0 ∅ = ∅"
    and "j : Ord   ⟹ Variants set (succ j) x = 𝒫 x"
    and "μ : Limit ⟹ Variants set μ f = ∅"
  using v_set_0 v_set_succ v_set_limit by auto

theorem tierI_mset :
  assumes j : "j : Ord" and x' : "x' : Set"
      and x'_sub: "x' ⊆ Tier j ⊖ set"
    shows "<set,x'> ∈ Tier (succ j)"
proof -
  have "x' ∈ 𝒫 (Tier j ⊖ set)" 
    by (rule powI[OF x' tier_ex_set[OF j] x'_sub])
  hence "x' ∈ Variants set (succ j) (Tier j ⊖ set)"
    unfolding variants_set(2)[OF j] by assumption
  thus "<set,x'> ∈ Tier (succ j)" 
    by (rule tier_succI2[OF set_tag j])
qed


theorem tierD_mset : 
  assumes j : "j : Ord" 
      and x : "<set,x'> ∈ Tier (succ j)"
    shows "x' ⊆ Tier j ⊖ set"
proof (rule tier_succE_pair[OF j x]) 
  show "<set,x'> ∈ Tier j ⟹ x' ⊆ Tier j ⊖ set"
  proof (induct rule: trans_induct3[OF j])
    case 1 then show ?case 
      using tier0D(2) variants_set(1) by auto
  next
    case IH:(2 j)
    show ?case 
    proof (rule tier_succE_pair[OF IH(1,3)])
      assume "<set,x'> ∈ Tier j"
      hence "x' ⊆ Tier j ⊖ set" using IH(2) by auto
      thus "x' ⊆ Tier (succ j) ⊖ set"  
        using subset_trans[OF _ tier_ex_subset_succ[OF IH(1)]] by auto
    next
      assume "x' ∈ Variants set (succ j) (Tier j ⊖ set)"
      hence "x' ⊆ Tier j ⊖ set" 
        unfolding variants_set(2)[OF IH(1)] 
        using powD[OF tier_ex_set[OF IH(1)]] by auto
      thus "x' ⊆ Tier (succ j) ⊖ set"  
        using subset_trans[OF _ tier_ex_subset_succ[OF IH(1)]] by auto
    qed
  next
    case IH:(3 μ)
    show ?case 
    proof (rule tier_limE_pair[OF ‹μ : Limit› ‹<set,x'> ∈ Tier μ›])
      fix j assume "j : Ord" "j < μ" "<set,x'> ∈ Tier j"
      hence "x' ⊆ Tier j ⊖ set" using IH(2) by auto 
      thus "x' ⊆ Tier μ ⊖ set" 
        using subset_trans[OF _ tier_limit_ex_subset[OF ‹μ : Limit› ‹j : Ord› ‹j < μ›]] by auto
    next
      show "x' ∈ Variants set μ (λj<μ. Tier j ⊖ set) ⟹ x' ⊆ Tier μ ⊖ set"
        unfolding variants_set(3)[OF ‹μ : Limit›] by auto
    qed
  qed
next
  assume "x' ∈ Variants set (succ j) (Tier j ⊖ set)"
  thus "x' ⊆ Tier j ⊖ set" 
    unfolding variants_set(2)[OF j] 
    using powD[OF tier_ex_set[OF j]] by auto
qed

theorem mI_mset :
  assumes j: "j : Ord" and x':"x' : Set"
      and x'_sub : "x' ⊆ Tier j ⊖ set"
  shows "<set,x'> : M"
  using mI[OF succ_ord[OF j]] tierI_mset[OF j x' x'_sub] by auto

theorem mE_mset : 
  assumes x : "<set,x'> : M"
  obtains j where "j : Ord" "x' ⊆ Tier j ⊖ set"
proof (rule m_cases_pair[OF x], use variants_set in auto)
  fix j assume "j : Ord" "x' ∈ 𝒫 (Tier j ⊖ set)"
  hence "j : Ord" "x' ⊆ Tier j ⊖ set"
    using powD[OF tier_ex_set] by auto
  thus ?thesis ..
qed
