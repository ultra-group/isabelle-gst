subsection ‹Operator for removing Excluded stuff from a set›

definition RemoveExcluded :: "'a ⇒ 'a ⇒ 'a" (infix ‹⊖› 60)
  where "RemoveExcluded x i ≡ { b ∈ x | ¬ Excluded i b }"

lemma exset_set : 
  assumes "x : Set" 
  shows "x ⊖ i : Set"
  unfolding RemoveExcluded_def 
  by (rule collect_set[OF assms])

lemma exset_iff : 
  assumes "x : Set" 
  shows "b ∈ x ⊖ i ⟷ b ∈ x ∧ ¬ Excluded i b"
  unfolding RemoveExcluded_def 
  using collect_iff[OF assms] by auto

lemma exsetI : 
  assumes x:"x : Set" 
      and b:"b ∈ x" "¬ Excluded i b"
    shows "b ∈ x ⊖ i"
  unfolding RemoveExcluded_def 
  using collectI[OF x] b by auto

lemma exsetD1 : 
  assumes x:"x : Set" and b:"b ∈ x ⊖ i"
  shows "b ∈ x"
  using exset_iff x b by auto

lemma exsetD2 : 
  assumes x:"x : Set" and b:"b ∈ x ⊖ i"
  shows "¬ Excluded i b"
  using exset_iff x b by auto

lemma exsetD_conj : 
  assumes x:"x : Set" and b:"b ∈ x ⊖ i"
  shows "b ∈ x ∧ ¬ Excluded i b"
  using exset_iff x b by auto

lemma exsetE :
  assumes x:"x : Set" and b:"b ∈ x ⊖ i"
  obtains "b ∈ x" "¬ Excluded i b"
  using exset_iff x b by auto


lemma ex_subset1 : 
  assumes x:"x : Set" and y:"y : Set"
      and sub:"x ⊆ y" 
    shows "a ⊆ x ⊖ i ⟹ a ⊆ y ⊖ i"
  using subset_iff exset_iff x y sub 
  by auto

lemma ex_subset2 : 
  assumes x:"x : Set" and y:"y : Set" 
    shows "x ⊆ y ⊖ i ⟹ x ⊆ y"
  using subset_iff exset_iff x y 
  by auto

  
section ‹Tier -- model building operator›

abbreviation Tier_zero :: 'a
  where "Tier_zero ≡ ⨄ (λi. Variants i 0 ∅)"

abbreviation Tier_succ :: "['a, 'a] ⇒ 'a"
  where "Tier_succ j x ≡ x ∪ ⨄ (λi. Variants i j (x ⊖ i))"

abbreviation Tier_lim :: "['a, 'a ⇒ 'a] ⇒ 'a"
  where "Tier_lim μ f ≡ (⋃ {f j | j < μ}) 
                      ∪ ⨄ (λi. Variants i μ (λj<μ. f j ⊖ i))"

definition Tier :: "'a ⇒ 'a" where
 "Tier ≡ OrdRec Tier_lim Tier_succ Tier_zero"

lemma tier_cases : 
  shows "Tier 0 = Tier_zero" 
    and "j : Ord ⟹ Tier (succ j) = Tier_succ (succ j) (Tier j)"
    and "μ : Limit ⟹ Tier μ = Tier_lim μ (λj. if j < μ then Tier j else OrdRec_default)"
  unfolding Tier_def 
  by (simp add: ordrec_0, use ordrec_succ ordrec_lim in auto)

qed

lemma tier_lim_set_ih : 
  assumes μ:"μ : Limit"
      and tμ : "⋀j. j : Ord ⟹ j < μ ⟹ Tier j : Set"  
    shows "Tier μ : Set"
  unfolding tier_cases(3)[OF μ]
proof (rule un_set[OF union_set])
  let ?L = "λj. if j < μ then Tier j else OrdRec_default"
  show "{ ?L j | j<μ } : SetOf Set"
  by (rule repfun_ord_setof[OF limit_ord[OF μ] mem_funI], 
      drule predsetD[OF limit_ord[OF μ]], use tμ in auto)
  show "⨄ (λi. Variants i μ (λj<μ. ?L j ⊖ i)) : Set"
  by (rule disj_set[OF variants_limit_setseq'[OF μ]])
qed

subsubsection ‹Tier 0›

lemma variants0_pair :
  assumes i : "i : Tag" and b' : "b' ∈ Variants i 0 ∅" 
    shows "<i,b'> : Pair"
  using pair_pair[OF tag_pmem[OF i] smem_pmem[OF setmemI]]
        variants_set[OF i zero_nonlimit emp_set] b' 
  by auto

lemma tier0_iff : 
  "b ∈ Tier 0 ⟷ (∃ i : Tag. ∃b' ∈ Variants i 0 ∅. b : Pair ∧ b = <i,b'>)"
  unfolding tier_cases(1) 
  by (rule disj_iff[OF variants_setseq1[OF zero_nonlimit emp_set]])

lemma tier0I :
  assumes i:"i : Tag" and b':"b' ∈ Variants i 0 ∅"
    shows "<i,b'> ∈ Tier 0" 
proof - 
  have "<i,b'> : Pair" 
    using pair_pair[OF tag_pmem[OF i(1)] smem_pmem[OF setmemI]]
          variants_set[OF i zero_nonlimit emp_set] b' by metis
  thus "<i,b'> ∈ Tier 0" unfolding tier0_iff using i b' by auto
qed

lemma tier0E :
  assumes b:"b ∈ Tier 0"
  obtains i b' where "i : Tag" "b' ∈ Variants i 0 ∅" "b : Pair" "b = <i,b'>" 
  using b unfolding tier0_iff by auto

lemma tier0D : 
  assumes b:"<i,b'> ∈ Tier 0"
  shows "i : Tag" "b' ∈ Variants i 0 ∅" "<i,b'> : Pair" 
proof -
  from tier0E[OF b] obtain j c' 
    where "j : Tag" "c' ∈ Variants j 0 ∅" "<j,c'> : Pair" "<i,b'> = <j,c'>" 
    by metis
  moreover hence "i = j" "b' = c'" using pair_inject by auto
  ultimately show "i : Tag" "b' ∈ Variants i 0 ∅" "<i,b'> : Pair" by auto
qed

subsubsection ‹Tier 'succ'›

lemma variants_succ_pair :
  assumes i : "i : Tag" and j : "j : Ord"
      and b' : "b' ∈ Variants i (succ j) (Tier j ⊖ i)" 
    shows "<i,b'> : Pair"
  using pair_pair[OF tag_pmem[OF i(1)] smem_pmem[OF setmemI]]
        variants_set[OF i succ_nonlimit[OF j] tier_ex_set[OF j]] b' 
  by auto

lemma tier_succ_iff :
  assumes j:"j: Ord" 
  shows "b ∈ Tier (succ j) ⟷ b ∈ Tier j ∨ 
          (∃i : Tag. ∃b'∈ Variants i (succ j) (Tier j ⊖ i). b : Pair ∧ b = <i,b'>)"
  unfolding tier_cases(2)[OF assms] 
            un_iff[OF tier_set[OF j] disj_set[OF variants_succ_setseq[OF j tier_set[OF j]]]] 
            disj_iff[OF  variants_succ_setseq[OF j tier_set[OF j]]]
  by rule

lemma tier_succI1 :
  assumes j : "j  : Ord" 
      and b : "b ∈ Tier j"
    shows "b ∈ Tier (succ j)"
  using tier_succ_iff[OF j] b by auto

lemma tier_succI2 :
  assumes i : "i : Tag" and j : "j : Ord"
    and b' : "b' ∈ Variants i (succ j) (Tier j ⊖ i)"
  shows "<i,b'> ∈ Tier (succ j)"
  using tier_succ_iff[OF j] i b' 
        variants_succ_pair[OF i j b'] 
  by auto

lemma tier_succE :
  assumes 
    j : "j : Ord" and b : "b ∈ Tier (succ j)"
  obtains (pred) "b ∈ Tier j" 
        | (fresh) i b' where "i : Tag" "b : Pair" "b = <i,b'>" 
                       "b' ∈ Variants i (succ j) (Tier j ⊖ i)" 
  using b unfolding tier_succ_iff[OF j] 
  by blast 

lemma tier_succE_pair :
  assumes
     j : "j : Ord" and b : "<i,b'> ∈ Tier (succ j)"
   obtains
     (pred)  "<i,b'> ∈ Tier j"
   | (fresh) "b' ∈ Variants i (succ j) (Tier j ⊖ i)"
  by (erule tier_succE[OF j b], metis pair_inject)

lemma tier_subset_succ :
  assumes j:"j : Ord"
  shows "Tier j ⊆ Tier (succ j)"
  using tier_succI1[OF j] by auto

lemma tier_ex_subset_succ :
  assumes j:"j : Ord"
  shows "Tier j ⊖ i ⊆ Tier (succ j) ⊖ i"
  using ex_subset1[OF tier_set[OF j] tier_set[OF succ_ord[OF j]] tier_subset_succ[OF j] subset_refl] 
  by auto

subsubsection ‹Tier 'limit'›

lemma variants_limit_pair :
  assumes i : "i : Tag" and μ : "μ : Limit"
      and b' : "b' ∈ Variants i μ (λj<μ. Tier j ⊖ i)" 
    shows "<i,b'> : Pair"
  using pair_pair[OF tag_pmem[OF i(1)] smem_pmem[OF setmemI]]
        variants_set2[OF i μ lam_tier_ex_fun[OF μ]] b'  
  by auto

lemma tier_lim_iff : 
  assumes μ:"μ : Limit"
  shows "b ∈ Tier μ ⟷ (∃j<μ. b ∈ Tier j) 
          ∨ (∃ i : Tag. ∃b'∈ Variants i μ (λj<μ. Tier j ⊖ i). b : Pair ∧ b = <i,b'>)"
  unfolding tier_cases(3)[OF μ] 
            un_iff[OF repfun_union_ord_set[OF limit_ord[OF μ] tier_restrict_replfun[OF μ]]
                      disj_set[OF variants_limit_setseq[OF μ]]]
 unfolding disj_iff[OF variants_limit_setseq[OF μ]]
           repfun_union[OF predset_set[OF limit_ord[OF μ]] tier_restrict_replfun[OF μ]] 
 unfolding lam_tier_restrict_eq[OF μ] tex_def            
           ord_ex_iff[OF limit_ord[OF μ]] by auto


lemma tier_limI1 :
  assumes μ : "μ : Limit" and j : "j : Ord" "j < μ"
      and b : "b ∈ Tier j"
    shows "b ∈ Tier μ"
  unfolding tier_lim_iff[OF μ]
  using b j by auto     

lemma tier_limI2 :
  assumes i : "i : Tag" and μ : "μ : Limit"
      and b : "b' ∈ Variants i μ (λj<μ. Tier j ⊖ i)"
    shows "<i,b'> ∈ Tier μ"
  unfolding tier_lim_iff[OF μ]
  using b i variants_limit_pair[OF i μ] by auto     

lemma tier_limE :
  assumes 
    μ : "μ : Limit" and 
    b : "b ∈ Tier μ"
  obtains (pred)  j where "j : Ord" "j < μ" "b ∈ Tier j" 
        | (fresh) i b' where "i : Tag" "b : Pair" "b = <i,b'>" 
                       "b' ∈ Variants i μ (λj<μ. Tier j ⊖ i)" 
  using b unfolding tier_lim_iff[OF μ] 
  by blast 

lemma tier_limE_pair :
  assumes 
    μ : "μ : Limit" and 
    b : "<i,b'> ∈ Tier μ"
  obtains (pred)  j where "j : Ord" "j < μ" "<i,b'> ∈ Tier j" 
        | (fresh) "b' ∈ Variants i μ (λj<μ. Tier j ⊖ i)" 
  by (erule tier_limE[OF μ b], metis pair_inject)


lemma tier_limit_union :
  assumes μ:"μ : Limit"
  shows "(⋃ j<μ. Tier j) ⊆ Tier μ"
proof 
  fix a assume "a ∈ (⋃ j<μ. Tier j)"  
  then obtain j where "j : Ord" "j < μ" "a ∈ Tier j"
    using repfun_union[OF lim_predset_set[OF μ] tier_replfun[OF μ]]
          predsetD[OF limit_ord[OF μ]] by auto
  thus "a ∈ Tier μ" 
    unfolding tier_lim_iff[OF μ]
    by auto
qed

lemma tier_limit_subset : 
  assumes μ : "μ : Limit" 
      and j : "j : Ord" "j < μ"
    shows "Tier j ⊆ Tier μ" 
    using repfun_union_subset[OF lim_predset_set[OF μ] 
                                 tier_set[OF j(1)] tier_replfun[OF μ]
                                 predsetI[OF j(1) limit_ord[OF μ] j(2)] subset_refl]
           subset_trans[OF _ tier_limit_union[OF μ]] 
    by auto

lemma tier_limit_ex_subset : 
  assumes μ : "μ : Limit" 
      and j : "j : Ord" "j < μ"
    shows "Tier j ⊖ i ⊆ Tier μ ⊖ i" 
  using ex_subset1[OF tier_set[OF j(1)] tier_set[OF limit_ord[OF μ]]
                      tier_limit_subset[OF assms] subset_refl] 
  by auto

lemma tier_increasing : 
  assumes i:"i : Ord" and j:"j : Ord"
  shows "i < j ⟹ Tier i ⊆ Tier j"
proof (induct rule: trans_induct3[OF j])
  case 1
  then show ?case 
  using zero_lt[OF i] by auto
next
  case IH:(2 j) 
  show ?case
  proof (rule leqE[OF i ‹j : Ord› ‹i ≤ j›])
    assume "i < j" 
    hence "Tier i ⊆ Tier j" by (rule IH(2))
    thus "Tier i ⊆ Tier (succ j)"
      by (rule subset_trans[OF _ tier_subset_succ[OF ‹j : Ord›]])
  next
    assume "i = j"
    thus "Tier i ⊆ Tier (succ j)"
      using tier_subset_succ[OF i] by auto
  qed
next
  case IH:(3 μ)
  show ?case using tier_limit_subset[OF IH(1) i IH(3)] by simp
qed

lemma greatest_tier : 
  assumes i : "i : Ord" and j : "j : Ord"
      and b : "b ∈ Tier i" and c : "c ∈ Tier j"
  obtains k where "k : Ord" "b ∈ Tier k" "c ∈ Tier k"
proof (rule linear[OF i j])
  assume "i < j" 
  hence "j : Ord" "b ∈ Tier j" "c ∈ Tier j"
    using tier_increasing[OF i j] b c j by auto
  thus ?thesis ..
next
  assume "i = j"
  hence "j : Ord" "b ∈ Tier j" "c ∈ Tier j"
    using j b c by auto
  thus ?thesis ..
next
  assume "j < i"
  hence "i : Ord" "b ∈ Tier i" "c ∈ Tier i"
    using tier_increasing[OF j i] b c i by auto
  thus ?thesis ..  
qed

lemma tier_typ_setof_depsum : 
  "Tier : Ord → SetOf (Σ i : Tag. α i)"
proof (rule funI, rule setofI[OF tier_set], auto)
  fix j b assume "j : Ord" thus "b ∈ Tier j ⟹ b : (Σ i : Tag. α i)"
  proof (induct j rule: trans_induct3)
    case zero show ?case 
    proof (rule tier0E[OF zero], auto, rule depsumI, auto)
      fix i b' 
      assume "i : Tag" and "b' ∈ Variants i 0 ∅" 
      thus "b' : α i" by (rule setof_mem[OF variants_setof[OF _ zero_nonlimit emp_set]])
    qed
  next
    case (succ j)
    show ?case 
    proof (rule tier_succE[OF succ(1,3) succ(2)], auto, 
           rule depsumI, auto)
      fix i b'
      assume "i : Tag" and "b' ∈ Variants i (succ j) (Tier j ⊖ i)"
      thus "b' : α i" 
        using setof_mem[OF variants_setof[OF _ succ_nonlimit tier_ex_set]] ‹j : Ord› by auto
    qed
  next
    case (lim μ)
    show ?case 
    proof (rule tier_limE[OF ‹μ : Limit› ‹b ∈ Tier μ›], use lim(2) in auto,
           rule depsumI, auto)
      fix i b' 
      assume "i : Tag" "b' ∈ Variants i μ (λj<μ. Tier j ⊖ i)"
      thus "b' : α i"
        using setof_mem[OF variants_setof2[OF _ lim(1) lam_tier_ex_fun[OF lim(1)]]] by auto
    qed
  qed
qed

     thus ?thesis ..
   qed
qed
