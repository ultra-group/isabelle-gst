subsection ‹ M - The soft type of all domain objects in a model›

definition inModel :: ‹'a ⇒ bool› (‹M›)
  where [typdef] : ‹inModel ≡ (λb. ∃i : Ord. b ∈ Tier i)›

lemma mI :
  assumes i : "i : Ord" 
      and b : "b ∈ Tier i"
    shows "b : M"
  unfolding inModel_def 
  by (rule tyI, rule texI, rule i, rule b)

lemma mE :
  assumes b : "b : M" 
  obtains i 
    where "i : Ord" "b ∈ Tier i"
  using b 
  unfolding inModel_def tex_def has_ty_def
  by auto

lemma m_induct :
  assumes x : ‹x : M› 
      and zero : ‹x ∈ Tier 0 ⟹ P›
      and succ : ‹⋀j. j : Ord ⟹ (x ∈ Tier j ⟹ P) ⟹ x ∈ Tier (succ j) ⟹ P›
      and lim  : ‹⋀μ. μ : Limit ⟹ (⋀j. j : Ord ⟹ j<μ ⟹ x ∈ Tier j ⟹ P) ⟹ x ∈ Tier μ  ⟹ P›
   shows "P"
proof (rule mE[OF x])   
   fix i assume "i : Ord" "x ∈ Tier i" 
   thus ?thesis 
    by (induct rule: trans_induct3, 
        use zero succ lim in auto)
qed

theorem m_depsum : 
  "x : M ⟹ x : (Σ i : Tag. α i)"  
  using tier_mem_depsum by (auto elim: mE)

lemmas mtagE = depsumE[OF m_depsum]
lemmas mtagD = depsumD[OF m_depsum]
lemmas mtagD_pair = depsumD_pair[OF m_depsum]
lemmas m_pair = mtagD(1)
lemmas mpair_pair = mtagD_pair(1)
lemmas mpair_inject = pair_inject[OF mpair_pair mpair_pair]

lemma mpair_snd_eq : 
  "<i,x'> : M ⟹ π <i,x'> = x'"
  using snd_eq[OF m_pair] by auto

theorem m_cases_pair : 
  assumes b : "<i,b'> : M"
  obtains 
    (zero) "b' ∈ Variants i 0 ∅"
  | (succ) j where "j : Ord" "b' ∈ Variants i (succ j) (Tier j ⊖ i)"
  | (limit) μ where "μ : Limit" "b' ∈ Variants i μ (λj<μ. Tier j ⊖ i)"      
proof (induct rule: m_induct[OF b])
  case 1
  hence "b' ∈ Variants i 0 ∅" using tier0D(2) by auto
  then show ?case ..
next
  case IH_succ: (2 j)
  show ?case 
  proof (rule tier_succE_pair[OF IH_succ(1,3)])
    assume "<i,b'> ∈ Tier j"
    thus ?case using IH_succ(2)[OF _ IH_succ(4,5,6)] by auto
  next
    assume "b' ∈ Variants i (succ j) (Tier j ⊖ i)"
    thus ?case using IH_succ(5)[OF ‹j : Ord›] by auto
  qed
next
  case IH_lim:(3 μ)
  show ?case 
  proof (rule tier_limE_pair[OF IH_lim(1,3)])
    fix j assume "j : Ord" "j < μ" "<i,b'> ∈ Tier j"
    thus ?case using IH_lim(2)[OF _ _ _] IH_lim(4,5,6) by auto
  next
    assume "b' ∈ Variants i μ (λj < μ. Tier j ⊖ i)"
    thus ?case using IH_lim(6)[OF ‹μ : Limit›] by auto
  qed
qed

lemmas tag_set_pair = pair_pair[OF ord_pmem[OF tag_ord] smem_pmem[OF set_setmem]]


subsection ‹Binders restricted to the model›
(*TODO: a 'Binder class parameterised by a restriction,

definition mall :: ‹('a ⇒ bool) ⇒ bool› (binder ‹m∀› 10)
  where "m∀x. P x ≡ ∀x : M. P x"

definition mex :: ‹('a ⇒ bool) ⇒ bool› (binder ‹m∃› 10)
  where "m∃x. P x ≡ ∃x : M. P x"

definition mex1 :: ‹('a ⇒ bool) ⇒ bool› (binder ‹m∃!› 10)
  where "m∃!x. P x ≡ ∃!x : M. P x"

definition muniq :: ‹('a ⇒ bool) ⇒ bool› (binder ‹m∃⇩≤⇩1› 10)
  where "m∃⇩≤⇩1x. P x ≡ ∃⇩≤⇩1x : M. P x"

definition mdefdes :: ‹('a ⇒ bool) ⇒ 'a ⇒ 'a›
  where "mdefdes P d ≡ ι x : M. P x else d"

lemma mallI [intro] : 
  assumes "⋀x. x : M ⟹ P x"
  shows "m∀x. P x"
  unfolding mall_def 
  using tallI assms by auto 

lemma mexI [intro] : 
  assumes "x : M" "P x"
  shows "m∃x. P x"
  unfolding mex_def 
  using texI assms by auto 


subsection ‹Type restricted model binders›

definition mtall :: ‹['a ⇒ bool, 'a ⇒ bool] ⇒ bool›
  where "mtall P Q ≡ m∀x. x : P ⟶ Q x"

definition mtex :: ‹['a ⇒ bool, 'a ⇒ bool] ⇒ bool›
  where "mtex P Q ≡ m∃x. x : P ∧ Q x"

definition mtex1 :: ‹['a ⇒ bool, 'a ⇒ bool] ⇒ bool›
  where "mtex1 P Q ≡ m∃!x. x : P ∧ Q x"

definition mtuniq :: ‹['a ⇒ bool, 'a ⇒ bool] ⇒ bool›
  where "mtuniq P Q ≡ m∃⇩≤⇩1 x. x : P ∧ Q x"

definition mtdefdes :: ‹['a ⇒ bool, 'a ⇒ bool, 'a] ⇒ 'a›
  where "mtdefdes P Q d ≡ mdefdes (λx. x : P ∧ Q x) d"

subsection ‹Model Rank operator›

definition mrank where "mrank x ≡ (least β. x ∈ Tier β)"

lemma mrank_tier : 
  assumes "x : M" 
  shows "x ∈ Tier (mrank x)"
proof (rule mE[OF assms])
  fix i assume "i : Ord" "x ∈ Tier i"
  thus "x ∈ Tier (mrank x)" unfolding mrank_def by (rule leastI)
qed

lemma mrank_eq : 
  assumes "j : Ord" 
  shows "x ∈ Tier j ⟹ (⋀i. i < j ⟹ x ∉ Tier i) ⟹ mrank x = j"
  unfolding mrank_def using least_eq[OF assms] by auto

lemma mrank_ord : 
  assumes x : "x : M"
  shows "mrank x : Ord" 
  unfolding mrank_def 
  using least_ord mE[OF x] by metis

end

subsection ‹Syntax for type-restricted model quantifiers›
(*Equivalent to soft-type restriction on quantifiers.
  Only purpose of this notation is for less keystrokes.*)
syntax
  "_mtall"  :: "[pttrn, 'a ⇒ bool, bool] ⇒ bool"  ("(3m∀_ : _./ _)" 10)
  "_mtex"   :: "[pttrn, 'a ⇒ bool, bool] ⇒ bool"  ("(3m∃_ : _./ _)" 10)
  "_mtex1"  :: "[pttrn, 'a ⇒ bool, bool] ⇒ bool"  ("(3m∃!_ : _./ _)" 10)
  "_mtuniq" :: "[pttrn, 'a ⇒ bool, bool] ⇒ bool"  ("(3m∃⇩≤⇩1_ : _./ _)" 10)
  "_mtdefdes" :: "[pttrn, 'a ⇒ bool, 'a ⇒ bool, 'a] ⇒ 'a"  
    ("(4mι _ : _./ _ else _)" 10)
translations
  "m∀x : P. Q"  ⇌ "CONST mtall P (λx. Q)"
  "m∃x : P. Q"  ⇌ "CONST mtex P (λx. Q)"
  "m∃!x : P. Q" ⇌ "CONST mtex1 P (λx. Q)"
  "m∃⇩≤⇩1x : P. Q" ⇌ "CONST mtuniq P (λx. Q)"
  "mι x : P. Q else d" ⇌ "CONST mtdefdes P (λx. Q) d"
