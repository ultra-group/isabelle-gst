Theory OrdQuants

theory OrdQuants 
  imports Ordinal
begin

section ‹Restricted quantification on the ordinal part of the < relation›

syntax
  "_oall" :: "[pttrn, 'a, bool] ⇒ bool"  (‹(3∀_<_./ _)› 10)
  "_oex"  :: "[pttrn, 'a, bool] ⇒ bool"  (‹(3∃_<_./ _)› 10)
translations
  "∀i<j. P" ⇀ "∀i:(CONST Ord). i < j ⟶ P"
  "∃i<j. P" ⇀ "∃i:(CONST Ord). i < j ∧ P"


context Ordinal begin

lemma oallI [intro!] : 
  "(⋀i. i : Ord ⟹ i < j ⟹ Q i) ⟹ ∀i < j. Q i"
   by auto

lemma tallE [elim!]: 
  assumes "∀i < j. Q i"
  obtains "⋀i. (i : Ord ⟹ i < j ⟹ Q i)"
  using assms by auto

lemma oallD [elim]: 
  "⟦ ∀i < j. Q i ; i : Ord; i < j ⟧ ⟹ Q i"
  by auto

lemma oexI [intro] : 
  "⟦ i : Ord ; i < j ; Q i ⟧ ⟹ ∃i < j. Q i"
  by auto

lemma oexE [elim!] : 
  assumes "∃i < j. Q i" 
  obtains i where "i : Ord" "i < j" "Q i"
  using assms by auto
end
end
