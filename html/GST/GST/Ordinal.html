subsection ‹Well-Ordering properties of < ›

lemma trans :
  assumes "i : Ord" "j : Ord" "k : Ord" 
      and "i < j" "j < k"
    shows "i < k"
  using lt_trans assms
  by auto

lemma asym :
  assumes "i : Ord" "j : Ord"
      and "i < j" "¬ P ⟹ j < i"
    shows "P"
  using lt_notsym assms
  by auto

lemma lt_reflE : 
  assumes "i : Ord" 
    shows "i < i ⟹ P"
  using asym assms 
  by auto

corollary not_refl :
  assumes "i : Ord"
    shows "¬ i < i"
  using lt_reflE[OF assms] 
  by auto

lemma lt_neq :
  assumes "i : Ord" "j : Ord"
  shows "i < j ⟹ i ≠ j"
  using lt_reflE assms 
  by auto

lemma linear : 
  assumes "i : Ord" "j : Ord"
  obtains (lt) "i < j" | (eq)  "i = j" | "j < i"
  using assms lt_linear
  by auto

subsection ‹succ - Successor operation›

lemma succ_lt :
  assumes "i : Ord"
    shows "i < succ i"
  using succ_ax assms by auto

lemma succ_neq : 
  assumes "i : Ord"
  shows "i ≠ succ i"
  by (rule lt_neq[OF assms succ_ord[OF assms] succ_lt[OF assms]])

lemma succ_nonzero :
  assumes "i : Ord"
  shows "succ i ≠ 0"
  using succ_lt zero_lt assms by auto

subsection ‹leq - Less Than or Equal To›

abbreviation leq (infixl ‹≤› 50) where "x ≤ y ≡ x < succ y"
lemma leq_iff : 
  assumes "i : Ord" "j : Ord"
    shows "i ≤ j ⟷ i < j ∨ i = j"
  using succ_ax assms unfolding tall_def 
  by auto

lemma leqI1 :
  assumes "i : Ord" "j : Ord"
    shows "i < j ⟹ i ≤ j"
  using leq_iff[OF assms] 
  by auto

lemma leqI2 :
  assumes "i : Ord" "j : Ord"
    shows "i = j ⟹ i ≤ j"
  using leq_iff[OF assms] 
  by auto
    
lemma leqE :
  assumes "i : Ord" "j : Ord" "i ≤ j"
      and "i < j ⟹ P" "i = j ⟹ P"
    shows "P"
  using leq_iff assms 
  by auto

lemma leq_refl :
  assumes "i : Ord"
    shows "i ≤ i"
  using leqI2 assms 
  by auto

lemma linear_lt_leq :
  assumes "i : Ord" "j : Ord"
  obtains (lt) "i < j" | (ge) "j ≤ i"
  by (rule linear[OF assms], use leqI1 leqI2 assms in auto)

lemma linear_leq :
  assumes "i : Ord" "j : Ord"
  obtains (le) "i ≤ j" | (ge) "j ≤ i"
  by (rule linear_lt_leq[OF assms], 
      blast intro: leqI1 leqI2 assms)

lemma not_lt_leq : 
  assumes "i : Ord" "j : Ord"
  shows "¬ i < j ⟹ j ≤ i"
  by (rule linear_lt_leq[OF assms], auto)

lemma leq_not_lt :
  assumes "i : Ord" "j : Ord"
    shows "j ≤ i ⟹ ¬ i < j"
  using leqE asym assms 
  by blast

lemma not_leq_iff_lt :
  assumes "i : Ord" "j : Ord"
    shows "¬ i < j ⟷ j ≤ i"
  using not_lt_leq leq_not_lt assms 
  by auto  

lemma neq_zero_lt :
  assumes "i : Ord"
    shows "i ≠ 0 ⟹ 0 < i"
proof -
  have "0 ≤ i" using zero_lt[OF assms] not_leq_iff_lt[OF assms zero_typ] by auto
  assume "i ≠ 0" 
  thus "0 < i" using leqE[OF zero_typ assms ‹0 ≤ i›] by auto
qed

lemma leqCI : 
  assumes "i : Ord" "j : Ord"
      and "i ≠ j ⟹ i < j"
    shows "i ≤ j"
  unfolding leq_iff[OF assms(1,2)] using assms(3) 
  by auto

lemma leq_antisym : 
  assumes "i : Ord" "j : Ord"
    shows "i ≤ j ⟹ j ≤ i ⟹ i = j"
  using leq_iff asym assms by auto

lemma leq_zero_iff :
  assumes "i : Ord"
    shows "i ≤ 0 ⟷ i = 0"
  unfolding leq_iff[OF ‹i : Ord› zero_typ] using zero_lt[OF ‹i : Ord›]
  by auto

lemma leq_zeroD :
  assumes "i : Ord"
    shows "i ≤ 0 ⟹ i = 0"
  by (rule leq_zero_iff[OF ‹i : Ord›, THEN iffD1])

lemma zero_leq :
  assumes "i : Ord"
    shows "0 ≤ i"
  using not_lt_leq[OF _ zero_typ] zero_lt assms
  by auto

lemma zero_lt_iff : 
  assumes "i : Ord"
    shows "i ≠ 0 ⟷ 0 < i"
  using zero_leq[OF assms] leq_iff[OF zero_typ assms] not_refl[OF zero_typ] by auto

lemma all_lt_leq : 
  assumes "i : Ord" "j : Ord"
      and lt: "⋀x. x : Ord ⟹ x < j ⟹ x < i"  
    shows "j ≤ i"
  using not_lt_leq not_refl assms 
  by auto

lemma leq_succ_iff :
  assumes "i : Ord" "j : Ord"
    shows "i ≤ succ j ⟷ i ≤ j ∨ i = succ j"
  unfolding leq_iff[OF ‹i : Ord› succ_ord[OF ‹j : Ord›]] 
  by rule



subsection ‹Transitivity Rules›

lemma lt_trans1 : 
  assumes "i : Ord" "j : Ord" "k : Ord"
  shows "i ≤ j ⟹ j < k ⟹ i < k"
  by (blast elim!: leqE[OF ‹i : Ord› ‹j : Ord›] 
            intro: trans[OF assms])

lemma lt_trans2 : 
  assumes "i : Ord" "j : Ord" "k : Ord"
  shows "i < j ⟹ j ≤ k ⟹ i < k"
  by (blast elim!: leqE[OF ‹j : Ord› ‹k : Ord›] 
            intro: trans[OF assms])

lemma leq_trans2 : 
  assumes "i : Ord" "j : Ord" "k : Ord"
  shows "i ≤ j ⟹ j ≤ k ⟹ i ≤ k"
  by (blast intro: lt_trans1[OF ‹i : Ord› ‹j : Ord› succ_ord[OF ‹k : Ord›]])

lemma succ_leqI :
  assumes "i : Ord" "j : Ord"
  shows "i<j ⟹ succ i ≤ j"
proof (rule not_leq_iff_lt[OF ‹j : Ord› succ_ord[OF ‹i : Ord›], THEN iffD1])
  assume "i < j" thus "¬ j ≤ i" 
    using not_leq_iff_lt[OF ‹i : Ord› ‹j : Ord›] by auto
qed

lemma succ_leqE : 
  assumes "i : Ord" "j : Ord"
  shows "succ i ≤ j ⟹ i < j"
proof - 
  assume "succ i ≤ j"
  hence "¬ j ≤ i" using not_leq_iff_lt[OF ‹j : Ord› succ_ord[OF ‹i : Ord›]] by auto
  thus "i < j" using not_leq_iff_lt[OF ‹i : Ord› ‹j : Ord›] by auto
qed

lemma succ_leq_iff : 
  assumes "i : Ord" "j : Ord"
  shows "succ i ≤ j ⟷ i < j"
  using succ_leqI succ_leqE assms
  by auto

lemma succ_leq_imp_leq : 
  assumes "i : Ord" "j : Ord"
  shows "succ i ≤ succ j ⟷ i ≤ j"
  using succ_leq_iff assms succ_ord 
  by auto

lemma lt_zero_lt :
  assumes "i : Ord" "j : Ord"
  shows "j < i ⟹ 0 < i"
  using lt_trans1[OF zero_typ ‹j : Ord› ‹i : Ord›]
        zero_leq[OF ‹j : Ord›] 
  by auto

lemma limit_ord : 
  assumes "μ : Limit"
    shows "μ : Ord"
  using assms unfolding Limit_def 
  by (unfold_typs)

lemma limit_lt_zero : 
  assumes "μ : Limit"
    shows "0 < μ"
  using assms unfolding Limit_def 
  by (unfold_typs)

lemma limit_nonzero : 
  assumes "μ : Limit"
  shows "μ ≠ 0"
  using zero_lt_iff[OF limit_ord] limit_lt_zero assms
  by auto

lemma limit_lt_succ :
  assumes "μ : Limit" "β : Ord" "β < μ"
    shows "succ β < μ"
  using assms unfolding Limit_def
  by (unfold_typs)

lemma limit_succE : 
  assumes i:"i : Ord" and succ_i:"succ i : Limit"
  shows "P"
proof (rule ccontr)
  have succ_i_ord:"succ i : Ord" by (rule limit_ord[OF succ_i])
  hence "¬ succ i < succ i" by (rule not_refl)
  thus "False" using limit_lt_succ[OF succ_i i leq_refl[OF i]] by auto
qed

definition nonLimit where [typdef] : "nonLimit ≡ Ord ¦ (λi. ¬ i : Limit)"

lemma nonLimitD :
  assumes "i : nonLimit"

  using assms by unfold_typs

lemmas nonLimitE = conjE[OF nonLimitD]
lemma nonlimit_ord :
  assumes i:"i : nonLimit"
  shows "i : Ord"

lemma limit_succ_lt_iff : 
  assumes μ:"μ : Limit" and i:"i : Ord"
  shows "succ i < μ ⟷ i < μ"
proof (rule)
  assume "succ i < μ"
  moreover have "i < succ i" by (rule succ_lt[OF i])
  ultimately show "i < μ" using trans[OF i succ_ord[OF i] limit_ord[OF μ]] by auto
next
  assume "i < μ" 
  thus "succ i < μ" by (rule limit_lt_succ[OF μ i]) 
qed

lemma not_zero_limit : "¬ 0 : Limit"

lemma limit_leq_succD :

lemma limitI : 
  assumes "i : Ord" 
    and "0 < i" "∀j : Ord. succ j ≠ i"
  shows "i : Limit" unfolding Limit_def 
proof (rule intI[OF ‹i : Ord›], rule tyI, rule, rule ‹0 < i›, rule, rule)
  fix j assume "j : Ord" "j < i"
  hence "¬ i ≤ j" "succ j ≠ i" using leq_not_lt[OF _ ‹i : Ord›] assms(3) by auto
  thus "succ j < i" using assms(3) linear[OF succ_ord[OF ‹j : Ord›] ‹i : Ord›] by auto
qed

lemma increasing_limitI : 
  assumes μ:"μ : Ord" and zero: "0 < μ" 
  and lt:"∀i : Ord. i < μ ⟶ (∃j : Ord. j < μ ∧ i < j)"
  shows "μ : Limit"
  unfolding Limit_def
proof (rule Soft_Types.intI[OF μ], rule tyI, rule conjI[OF zero], auto)
  fix i assume i:"i : Ord" and "i < μ" 
  then obtain j where j:"j : Ord" and "j < μ" "i < j" using lt by auto
  show "succ i < μ" by (rule lt_trans1[OF succ_ord[OF i] j μ succ_leqI[OF i j ‹i < j›] ‹j < μ›])
qed

subsection ‹Trichotomy of Ordinals, Transfinite Induction›

lemma ord_cases_disj :
  assumes "i : Ord"
  shows "i = 0 ∨ (∃j : Ord. i = succ j) ∨ i : Limit"
  using limitI neq_zero_lt assms 
  by auto 

lemma ord_cases : 
  assumes "i : Ord"
  obtains (zero) "i = 0" 
        | (succ) j where "j : Ord" "i = succ j" 
        | (limit) "i : Limit"
  using ord_cases_disj[OF assms]
  by auto

lemma trans_induct [consumes 1, case_names step] :
  assumes "i : Ord" "⋀j. j : Ord ⟹ (⋀k. k : Ord ⟹ k < j ⟹ P k) ⟹ P j" 
  shows "P i"
  using lt_induct assms unfolding tall_def 
  by auto 

lemma trans_induct3 [case_names zero succ lim, consumes 1] :
  assumes "i : Ord" 
     and zero: "P 0" 
     and succ: "⋀j. ⟦ j : Ord; P j ⟧ ⟹ P (succ j)"
     and lim : "⋀μ. ⟦ μ : Limit; ∀j : Ord. j < μ ⟶ P j ⟧ ⟹ P μ"
   shows "P i" 
proof (rule trans_induct[OF ‹i : Ord›])
  fix j assume j:"j : Ord" 
  and IH: "⋀k. k : Ord ⟹ k < j ⟹ P k"
  show "P j" proof (rule ord_cases[OF j])
    assume "j = 0" thus "P j" using zero by auto
  next
    fix k assume "k : Ord" "j = succ k"
    hence "k < j" using succ_lt by auto
    thus "P j" using IH[OF ‹k : Ord›] succ[OF ‹k : Ord›] ‹j = succ k› by auto
  next
    assume "j : Limit"
    thus "P j" using lim IH by auto
  qed
qed

lemma omega_zero : "0 < ω" by (rule limit_lt_zero[OF omega_typ])

lemma omega_succ :
  assumes i:"i : Ord" 
  shows "i < ω ⟹ succ i < ω"
  by (rule limit_lt_succ[OF omega_typ i])

lemma omega_lt_cases :
  assumes i:"i : Ord" and lt:"i < ω"
  obtains (zero) "i = 0" | (succ) j where "j : Ord" "i = succ j"
proof (rule ord_cases[OF i], auto, rule ccontr)
  assume "i : Limit" hence "ω < i" 
    using lt_neq[OF i omega_ord lt] omega_ax by auto
  thus "False" using lt asym[OF i omega_ord] by auto
qed

lemma omega_lt_not_limit :
  assumes i:"i : Ord" and lt:"i < ω" 
  shows "¬ i : Limit"
  by (rule omega_lt_cases[OF i lt],

subsection ‹Least Ordinal operator›

definition least :: "('a ⇒ bool) ⇒ 'a" (binder ‹least › 10)
  where "least x. P x ≡ ι i. i : Ord ∧ P i ∧ (∀j : Ord. j < i ⟶ ¬ P j) else Ordinal_default"

lemma least_eq :
  assumes i:"i : Ord" and "P i"
    and lt:"⋀j. j : Ord ⟹ j < i ⟹ ¬ P j"
  shows "(least i. P i) = i" unfolding least_def
proof (rule the_def_eq, use assms in auto)
  fix k assume k:"k : Ord" and "P k" and j:"∀j. j : Ord ⟶ j < k ⟶ ¬ P j"
  show "k = i" by (rule linear[OF k i], use i k ‹P i› ‹P k› lt j in auto)
qed

lemma least_ord : 
  assumes i:"i : Ord"
  shows "P i ⟹ (least i. P i) : Ord"
proof (induct rule: trans_induct[OF i])
  fix j assume j:"j : Ord" and "P j"
  and k:"⋀k. k : Ord ⟹ k < j ⟹ P k ⟹ (least i. P i) : Ord"
  show "(least i. P i) : Ord"
  proof (cases "(least i. P i) : Ord")
    case True 
    then show ?thesis .
  next
    case False
    hence "⋀k. k : Ord ⟹ k < j ⟹ ¬ P k" using k by auto
    hence "(least i. P i) = j" using k least_eq[OF j, of P, OF ‹P j›] by auto
    then show ?thesis using ‹j : Ord› by auto
  qed
qed


lemma leastI : 
  assumes i:"i : Ord"
  shows "P i ⟹ P (least i. P i)"
proof (induct rule: trans_induct[OF i])
  fix j assume j:"j : Ord" and "P j"
  and k:"⋀k. k : Ord ⟹ k < j ⟹ P k ⟹ P (least i. P i)"
  show "P (least i. P i)"
  proof (cases "P (least i. P i)")
    case True 
    then show ?thesis .
  next
    case False
    hence "⋀k. k : Ord ⟹ k < j ⟹ ¬ P k" using k by auto
    hence "(least i. P i) = j" using k least_eq[OF j, of P, OF ‹P j›] by auto
    then show ?thesis using ‹P j› by auto
  qed
qed

lemma least_leqI :
  assumes i:"i : Ord"
  shows "P i ⟹ (least i. P i) ≤ i"
proof (induct rule: trans_induct[OF i])

    have "(least i. P i) = j"

        show "False" using leqI1[OF l j lt_trans1[OF l ‹k : Ord› j pk ‹k < j›]] lj by auto
      qed
    qed
    then show ?thesis using leq_refl[OF j] by auto
  qed
qed

lemma lt_leastE :
  assumes i:"i : Ord" and "P i"
    and lt : "i < (least i. P i)"
  shows "Q"
  using lt_trans2[OF i least_ord[OF i, of P, OF ‹P i›] i lt least_leqI[OF i, of P, OF ‹P i›]]  
        not_refl[OF i]
  by auto

(*Easier to apply than leastI: conclusion has only one occurrence of P*)
lemma leastI2 :
  assumes i:"i : Ord" and "P i"
  and lt: "⋀j. j : Ord ⟹ P j ⟹ Q j"
  shows "Q (least j. P j)"
  by (rule lt[OF least_ord[of i P, OF i ‹P i›] 
                 leastI[of i P, OF i ‹P i›]])

  "¬ (∃i : Ord. P i) ⟹ (least i. P i) = Ordinal_default"
