Theory Ordinal_Model

theory Ordinal_Model
  imports mLimit
begin

context Ordinal_Model 
begin

ML ‹val mOrdinal = 
  mk_msig [] "mOrdinal" @{class Ordinal}
    ["mOrd", "mlt", "mzero", "msucc", "momega", 
     "mLimit", "Ordinal_Model_mdefault"] @{context}›

translate_axioms mOrdinal_axioms : mOrdinal
proof -
  show "m∀x. x : mOrd ⟶ msucc x : mOrd"
    using msucc_mord by auto
  show "m0 : mOrd"
    using mzero_mord .
  show "mω : mLimit"
    using momega_mlimit .
  show "m∀b : mOrd. ¬ b ≪ m0"
    using mzero_ax .
  show "m∀a : mOrd. m∀b : mOrd. (a ≪ msucc b) = (a ≪ b ∨ a = b)"
    using msucc_ax .
  show "m∀μ : mLimit. μ = mω ∨ mω ≪ μ"
    using mlimit_ax .
  show "m∀i : mOrd. m∀j : mOrd. m∀k : mOrd. i ≪ j ⟶ j ≪ k ⟶ i ≪ k"
    using mlt_trans_ax .
  show "m∀i : mOrd. m∀j : mOrd. i ≪ j ⟶ ¬ j ≪ i"  
    using mlt_notsym_ax .
  show "m∀i : mOrd. m∀j : mOrd. i ≪ j ∨ i = j ∨ j ≪ i"
    using mlt_linear_ax .
  show "∀P. (m∀i : mOrd. (m∀j : mOrd. j ≪ i ⟶ P j) ⟶ P i) ⟶ mtall mOrd P"
    using mlt_induct_ax .
  show "mLimit = (mOrd △ (λμ. m0 ≪ μ ∧ (m∀j : mOrd. j ≪ μ ⟶ msucc j ≪ μ)))"
    using mlimit_def_ax .
qed

resp_thms mOrdinal_rsp : mOrdinal
 by (rule mzero_m, rule msucc_m, rule momega_m, 
    rule Ordinal_Model_mdefault_m)

end
end
