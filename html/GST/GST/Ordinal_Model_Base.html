theory Ordinal_Model_Base
  imports "../../ModelKit/ModelComponents"
begin

context ModelBase begin
ML ‹val ord_model = mcomp 
 { name = "Ordinal_Model", 
   deps = [], 
   variant = SOME (vari 
    { tag_name = "ord",
      vcases = 
        (@{term ‹{0}›}, 
         @{term "(λj x. {j}) :: 'a ⇒ 'a ⇒ 'a"}, 
         @{term "(λμ f. {μ}) :: 'a ⇒ 'a ⇒ 'a"}),
      alpha_typ = @{term ‹Ord›},
      excludes_formulas = [@{prop ‹¬ Excluded ord <ord, j>›}]})} ›
end
local_setup ‹snd o (mk_mcomp_class ord_model)›

context Ordinal_Model 
begin
(*MOVE ME*)
lemmas zero_ord = zero_typ

definition mOrd :: ‹'a ⇒ bool›
  where "mOrd ≡ M △ ▸ ord"

definition mlt :: ‹'a ⇒ 'a ⇒ bool› (infix ‹≪› 50)
  where "i ≪ j ≡ i : mOrd ∧ j : mOrd ∧ snd i < snd j"

definition mzero :: ‹'a› (‹m0›)
  where "m0 ≡ <ord,0>"

definition msucc' :: ‹'a ⇒ 'a›
  where "msucc' j ≡ <ord, succ (snd j)>"

definition msucc :: ‹'a ⇒ 'a›
  where "msucc j ≡ if j : mOrd then msucc' j 
                               else Ordinal_Model_mdefault"

definition momega :: ‹'a› (‹mω›)
  where "mω ≡ <ord, ω>"

definition mLimit :: ‹'a ⇒ bool›
  where "mLimit j ≡ j : mOrd ∧ snd j : Limit"

theorem variants_ord_cases :
  shows "Variants ord 0 x = {0}"
    and "j : Ord   ⟹ Variants ord (succ j) x = {succ j}"
    and "μ : Limit ⟹ Variants ord μ f = {μ}"
  using v_ord_0 v_ord_succ v_ord_limit by auto

corollary variants_ord :
  assumes j : "j : Ord"
  shows "Variants ord j x = {j}"
  by (rule ord_cases[OF j], 
      use variants_ord_cases in auto)

corollary variants_ord_mem : 
  assumes j : "j : Ord"
  shows "i ∈ Variants ord j x ⟹ i = j"
  unfolding variants_ord[OF j]
     sng_iff[OF ord_setmem[OF j]] .
  
lemma ord_tierI_cases :
  shows "<ord, 0> ∈ Tier 0"
    and "j : Ord   ⟹ <ord,succ j> ∈ Tier (succ j)"
    and "u : Limit ⟹ <ord, u> ∈ Tier u"
 using tier0I[OF ord_tag] tier_succI2[OF ord_tag] tier_limI2[OF ord_tag]   
       variants_ord sng_iff[OF ord_setmem] zero_ord succ_ord limit_ord 
 by auto

lemma ord_tierI_eq :
  assumes i : "i : Ord"
    shows "<ord, i> ∈ Tier i"
  by (rule ord_cases[OF i], 
      use ord_tierI_cases in auto) 

lemma ord_tierI :
  assumes i : "i : Ord" and j : "j : Ord"
      and ij: "i ≤ j"
    shows "<ord, i> ∈ Tier j"
  by (rule leqE[OF i j ij],
      use ord_tierI_eq[OF i] tier_increasing[OF i j] in auto)
    
lemma ord_tierD : 
  assumes i : "i : Ord" and j : "j : Ord"
  shows "<ord,i> ∈ Tier j ⟹ i ≤ j"
proof (induct rule: trans_induct3[OF j])
  case 1
  hence "i ∈ Variants ord 0 ∅" 
    using tier0D(2) by simp
  hence "i = 0" 
    using variants_ord_mem[OF zero_ord] by simp
  then show ?case 
    using leq_zero_iff[OF i] by auto
next
  case ih:(2 j)
  show ?case 
  proof (rule tier_succE_pair[OF ih(1) ih(3)])
    assume "<ord,i> ∈ Tier j"
    hence "i ≤ j" using ih(2) by simp
    thus "i ≤ succ j" 
      using leq_succ_iff[OF i ih(1)] by auto
  next
    assume "i ∈ Variants ord (succ j) (Tier j ⊖ ord)"
    hence "i = succ j" 
      using variants_ord_mem[OF succ_ord[OF ih(1)]] by auto
    thus "i ≤ succ j" 
      using leqI2[OF i succ_ord[OF ih(1)]] by auto
  qed
next
  case ih:(3 μ)
  show ?case 
  proof (rule tier_limE_pair[OF ih(1) ih(3)])
    fix j 
    assume j:"j : Ord" "j < μ"
       and ij:"<ord,i> ∈ Tier j"
    hence "i ≤ j" 
      using ih(2) by auto 
    thus "i ≤ μ" 
      using leqI1[OF j(1) limit_ord[OF ih(1)] j(2)]
            leq_trans2[OF i j(1) limit_ord[OF ih(1)]] 
      by auto
  next
    assume "i ∈ Variants ord μ (λj<μ. Tier j ⊖ ord)"
    hence "i = μ" 
      using variants_ord_mem[OF limit_ord[OF ih(1)]] by auto
    thus "i ≤ μ" 
      by (rule leqI2[OF i limit_ord[OF ih(1)]])
  qed
qed      

lemma ord_tier_iff : 
  assumes i : "i : Ord" and j : "j : Ord"
    shows "<ord, i> ∈ Tier j ⟷ i ≤ j"
  using ord_tierI[OF i j] ord_tierD[OF i j] 
  by auto



end
