context GZF begin

text ‹Mapping a λ-function over a set›
definition RepFun :: "['a, 'a ⇒ 'a] ⇒ 'a" where
  "RepFun x F ≡ Repl x (λa b. b = F a)"

text ‹Filtering a set using a predicate›
definition Collect :: "'a ⇒ ['a ⇒ bool] ⇒ 'a" where
  "Collect x P ≡ Repl x (λa b. a = b ∧ P a)"

end

syntax
  "_RepFun" :: "['a, pttrn, 'a] => 'a"  (‹(1{_ |/ _ ∈ _})› [51,0,51])
  "_Collect" :: "[pttrn, 'a, bool ] ⇒ 'a"  (‹(1{_ ∈ _ |/ _})›)
translations
  "{c | b∈x}" ⇌ "CONST RepFun x (λb. c)"
  "{b∈x | P}" ⇌ "CONST Collect x (λb. P)"


context GZF begin

lemma any_replpred : 
 "x : Set ⟹ (λa b. b = F a) : ReplPred x"
 by (rule replpredI, unfold tuniq_def Uniq_def, auto)

lemma RepFun_typ_set : "RepFun : (Π x : Set. Any → Set)"
  unfolding RepFun_def 
  by (rule depfunI, rule funI, use any_replpred repl_set in auto)

definition mem_fun_ty :: ‹'a ⇒ ('a ⇒ bool) ⇒ (('a ⇒ 'a) ⇒ bool)› 
  (infixr ‹↝› 50) where [typdef] : ‹x ↝ P ≡ MemOf x → P›

lemma mem_funI : 
  assumes "⋀b. b ∈ x ⟹ F b : P"
    shows "F : x ↝ P"
  using assms 
  unfolding mem_fun_ty_def MemOf_def fun_ty_def has_ty_def
  by auto
     
lemma mem_funE : 
  assumes "F : x ↝ P" "b ∈ x"
    shows "F b : P"
  using assms 
  unfolding mem_fun_ty_def MemOf_def fun_ty_def has_ty_def
  by auto

lemma replfun_funpred : 
  assumes x : "x : Set" and F : "F : x ↝ β"  
    shows "(λa b. b = F a) : ReplFunPred x β"
  unfolding ReplFunPred_def
  by (rule intI, rule any_replpred[OF x], rule bpredI, 
      use memofD mem_funE[OF F] in auto)

lemma RepFun_typ : 
  "RepFun : (Π x : Set. (x ↝ β) → SetOf β)" 
proof (rule depfunI, rule funI)
  fix x :: 'a and F :: "'a ⇒ 'a" 
  assume "x : Set" "F : x ↝ β"
  hence "(λa b. b = F a) : ReplFunPred x β" 
    using replfun_funpred by auto
  thus "RepFun x F : SetOf β" 
    unfolding RepFun_def by (rule funE[OF depfunE[OF Repl_typ2 ‹x : Set›]])
qed

lemmas repfun_setof = funE[OF depfunE[OF RepFun_typ]]
lemmas repfun_set = funE[OF depfunE[OF RepFun_typ_set] anyI]

lemma repfunI : 
  assumes x : "x : Set"
      and b : "b ∈ x" and Fb : "F b : SetMem"
    shows "F b ∈ {F b | b ∈ x}" 
  unfolding RepFun_def
  by (rule replaceI[OF x any_replpred[OF x]], 
      use b Fb in auto)

lemma repfun_eqI : 
  assumes "x : Set"
    shows "⟦ c : SetMem ; c = F b ; b ∈ x ⟧ ⟹ c ∈ {F a | a ∈ x}"
  using repfunI[OF assms] by auto

lemma repfunE : 
  assumes x : "x : Set" and b : "b ∈ RepFun x F"
  obtains a where "a ∈ x" "b : SetMem" "b = F a" 
    using b unfolding RepFun_def
    using replaceE[OF x any_replpred[OF x], of b F] by blast
    
lemma repfun_cong : 
  assumes x : "x : Set" and y : "y : Set"
  shows "⟦ x = y ; ⋀a. a ∈ y ⟹ F a = G a ⟧ ⟹ RepFun x F = RepFun y G" 
  unfolding RepFun_def 
  using replace_cong[OF x any_replpred[OF x] y any_replpred[OF y]] by simp 

lemma repfun_iff  : 
  assumes "x : Set"
  shows "b ∈ RepFun x F ⟷ (∃c ∈ x. b = F c) ∧ b : SetMem"
  using repfun_eqI[OF assms] repfunE[OF assms] by blast

lemma repfun_union : 
  assumes x : "x : Set" and F : "F : x ↝ Set"
  shows "b ∈ ⋃ {F a | a ∈ x} ⟷ (∃a ∈ x. b ∈ F a)"
proof - 
  have R:"RepFun x F : SetOf Set" 
    by (rule funE[OF depfunE[OF RepFun_typ x] F])
  show ?thesis 
    unfolding union_iff[OF R] bex_def rex_def 
              repfun_iff[OF ‹x : Set›]
    by (auto, use set_setmem[OF mem_funE[OF F]] in auto )   
qed

lemma repfun_union_subset : 
  assumes x : "x : Set" and y : "y : Set" 
      and F : "F : x ↝ Set" 
    shows "a ∈ x ⟹ y ⊆ F a ⟹ y ⊆ ⋃ {F a | a ∈ x}"
proof -
  assume "a ∈ x" "y ⊆ F a"
  hence "F a : Set" using mem_funE[OF F] by auto
  have "⋃ RepFun x F : Set" 
    using union_set[OF repfun_setof[OF x F]] .
  from ‹a ∈ x› ‹y ⊆ F a› show ?thesis 
    unfolding repfun_union[OF x F] subset_iff
    by auto
qed


subsection ‹Rules for subset comprehension›

lemma collect_replpred : 
  "x : Set ⟹ (λa b. a = b ∧ P a) : ReplPred x"
  using setmemI
  unfolding ReplPred_def tuniq_def Uniq_def has_ty_def
  by auto
      (* rule, rule, rule, drule setmemI[OF ‹x : Set›], unfold has_ty_def, auto *)

lemma Collect_typ : "Collect : Set → Any → Set"
proof (rule funI)+
  fix x and P :: "'a => bool" assume "x : Set"
  moreover hence "(λa b. a = b ∧ P a) : ReplPred x" by (rule collect_replpred)
  ultimately show "Collect x P : Set" unfolding Collect_def by (rule repl_set)
qed

lemmas collect_set = funE[OF funE[OF Collect_typ] anyI]

lemma collect_iff : 
  assumes x : "x : Set"
    shows "a ∈ { a ∈ x | P a} ⟷ a ∈ x ∧ P a"
  unfolding Collect_def replace_iff[OF ‹x : Set› collect_replpred[OF ‹x : Set›]] 
  using setmemI[OF x] by auto

lemma collectI : 
  assumes "x : Set" 
    shows "⟦ a ∈ x ; P a ⟧ ⟹ a ∈ { a ∈ x | P a }"
  using collect_iff[OF ‹x : Set›] by auto

lemma collectE : 
  assumes "x : Set" 
    shows "⟦ a ∈ { a ∈ x | P a } ; ⟦ a ∈ x ; P a ⟧ ⟹ R ⟧ ⟹ R"
  using collect_iff[OF assms] by auto

lemma collectD1 : 
  assumes "x : Set" 
    shows "a ∈ { a∈x | P a } ⟹ a ∈ x"
  using collectE[OF assms] by auto

lemma collectD2 : 
  assumes "x : Set" 
    shows "a ∈ { a∈x | P a } ⟹ P a"
  using collectE[OF assms] by auto

lemma collect_subset : 
  assumes "x : Set" 
    shows "{ a∈x | P a } ⊆ x"
proof (rule subsetI)
  fix a assume "a ∈ Collect x P" 
  thus "a ∈ x" by (rule collectD1[OF ‹x : Set›])
qed

lemma collect_typ_subset : 
  assumes "x : SetOf α" 
  shows "{ a∈x | P a } : SetOf α"
proof (rule setof_subset[OF _ ‹x : SetOf α›])
  have "x : Set" by (rule setof_set[OF ‹x : SetOf α›])
  thus "Collect x P : Set" and "Collect x P ⊆ x" 
    by (rule collect_set, rule collect_subset)
qed

lemma Collect_cong [cong]: 
  assumes "x : Set" "y : Set" 
  shows [cong]: "⟦ x = y;  ⋀a. a ∈ y ⟹ P a ⟷ Q a ⟧ 
                  ⟹ { a∈x | P a } = { a∈y | Q a }"
  unfolding Collect_def using collect_replpred[OF ‹x : Set›, of P] 
                              collect_replpred[of _ Q] assms by auto

end
