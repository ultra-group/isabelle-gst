subsection ‹Typing judgements›
definition has_ty :: "'a ⇒ ('a ⇒ bool) ⇒ bool"  (infix ":" 45)
  where "x : P ≡ P x"

lemma tyI : "P x ⟹ x : P" unfolding has_ty_def by simp
lemma tyE : "x : P ⟹ P x" unfolding has_ty_def by simp 


definition tall :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool"
  where "tall P Q ≡ ∀x. x : P ⟶ Q x"

definition tex :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool"
  where "tex P Q ≡ ∃x. x : P ∧ Q x"

definition tex1 :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool"
  where "tex1 P Q ≡ ∃!x. x : P ∧ Q x"

definition tuniq :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool"
  where "tuniq P Q ≡ ∃⇩≤⇩1x. x : P ∧ Q x"

definition typ_defdes :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ 'a ⇒ 'a"
  where "typ_defdes P Q d ≡ (ι x. x : P ∧ Q x else d) "

syntax
  "_tball"  :: "[pttrn, 'a ⇒ bool, bool] ⇒ bool"  ("(3∀_ : _./ _)" 10)
  "_tbex"   :: "[pttrn, 'a ⇒ bool, bool] ⇒ bool"  ("(3∃_ : _./ _)" 10)
  "_tex1"   :: "[pttrn, 'a ⇒ bool, bool] ⇒ bool"  ("(3∃!_ : _./ _)" 10)
  "_tuniq"  :: "[pttrn, 'a ⇒ bool, bool] ⇒ bool"  ("(3∃⇩≤⇩1_ : _./ _)" 10)
  "_tdefdes" :: "[pttrn, 'a ⇒ bool, 'a ⇒ bool, 'a] ⇒ 'a"  ("(4ι_ : _./ _ else _)" 10)
translations
  "∀x : A. P"  ⇌ "CONST tall A (λx. P)"
  "∃x : A. P"  ⇌ "CONST tex A (λx. P)"
  "∃!x : A. P" ⇌ "CONST tex1 A (λx. P)"
  "∃⇩≤⇩1x: A. P" ⇌ "CONST tuniq A (λx. P)"
  "ι x : A. P else default" ⇌ "CONST typ_defdes A (λx. P) default"


lemma tallI [intro!] : 
  "(⋀x. x : P ⟹ Q x) ⟹ ∀x : P. Q x"
  unfolding tall_def by auto

lemma tallE [elim!]: 
  assumes "∀x : P. Q x"
  obtains "⋀x. (x : P ⟹ Q x)"
  using assms unfolding tall_def by auto

lemma tallD [elim]: 
  "⟦ ∀x : P. Q x ; x : P ⟧ ⟹ Q x"
  by auto

lemma texI [intro] : 
  "⟦ x : P ; Q x ⟧ ⟹ ∃x:P. Q x"
  unfolding tex_def by auto

lemma texE [elim!] : 
  assumes "∃x:P. Q x" 
  obtains x where "x : P" and "Q x"
  using assms unfolding tex_def by auto

lemma typ_defdes_eq :
  assumes "b : A" "P b" 
      and eq:"⋀c. c : A ⟹ P c ⟹ c = b"
    shows "(ι b : A. P b else d) = b"
  unfolding typ_defdes_def 
  by (rule the_def_eq, rule conjI[OF ‹b : A› ‹P b›], 
      use assms(3) in auto)

lemma typ_defdes_eq' : 
  assumes ex1:"∃! b : A. P b" 
     and c:"c : A" "P c"
  shows "(ι b : A. P b else d) = c"
  using ex1 c unfolding tex1_def
  by (blast intro: typ_defdes_eq)

lemma typ_defdesI :
  assumes b:"b : A" "P b" 
      and eq:"⋀c. c : A ⟹ P c ⟹ c = b"
  shows "P (ι x : A. P x else d)"
proof -
  from typ_defdes_eq[of b A P, OF b] eq 
  have "(ι x : A. P x else d) = b" by auto  
  thus "P (ι x : A. P x else d)" using ‹P b› by simp 
qed

lemma typ_defdesI' : 
  assumes ex1:"∃! b : A. P b" 
  shows "P (ι b : A. P b else d)"
  using ex1 unfolding tex1_def
  by (blast intro: typ_defdesI)

  unfolding fun_ty_def has_ty_def by auto

  unfolding fun_ty_def has_ty_def by auto

subsection ‹Dependent Function types›

definition depfun_ty :: "('a ⇒ bool) ⇒ ('a ⇒ ('b ⇒ bool)) ⇒ (('a ⇒ 'b) ⇒ bool)" 
  where [typdef]: "depfun_ty P Q ≡ (λf. ∀x : P. f x : Q x)" 
syntax
  "_depfunty" :: "[pttrn, 'a ⇒ bool, bool] => bool" (‹(3Π _:_./ _)› 10)
translations
  "Π x:P. B" ⇌ "CONST depfun_ty P (λx. B)"

lemma depfunty_eq : 
  "f : depfun_ty P Q ≡ ∀x. x : P ⟶ f x : Q x" 
  unfolding depfun_ty_def tall_def has_ty_def by auto

lemma depfunI [typ_intro] : 
  "(⋀x. x : A ⟹ f x : B x) ⟹ f : (Π x : A. B x)"
  unfolding depfun_ty_def tall_def has_ty_def by auto 

lemma depfunE : 
  assumes "f : (Π x : A. B x)" "x : A"
  shows "f x : B x"
  using assms unfolding depfun_ty_def tall_def has_ty_def by auto

subsection ‹Intersection types›
definition inter_ty :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ ('a ⇒ bool)" (infix "△" 45)
  where [typdef] : "P △ Q ≡ λx. P x ∧ Q x"

lemma intE : 
  assumes "x : (P △ Q)" 
    shows "x : P ∧ x : Q" 
  using assms unfolding inter_ty_def by unfold_typs

lemma intE1 : "x : (P △ Q) ⟹ x : P" 
  unfolding inter_ty_def by unfold_typs

lemma intE2 : "x : (P △ Q) ⟹ x : Q" 
  unfolding inter_ty_def by unfold_typs

  "⟦ x : P ; x : Q ⟧ ⟹ x : (P △ Q)" 
  unfolding inter_ty_def by unfold_typs

subsection ‹Subtyping›
definition subtyp :: "('a ⇒ bool) ⇒ ('a ⇒ bool) ⇒ bool" (infix "<<" 45) 
  where "P << Q ≡ ∀x. x : P ⟶ x : Q"

lemma subtypI  : "(⋀x. x : P ⟹ x : Q) ⟹ P << Q" unfolding subtyp_def by (auto)
lemma subtypE : "P << Q ⟹ x : P ⟹ x : Q" unfolding subtyp_def by auto

lemma subtyp_trans : 
  assumes "P << Q" "Q << R"
  shows "P << R"
  by (rule subtypI, rule subtypE[OF ‹Q << R›], rule subtypE[OF ‹P << Q›], assumption)

lemma int_sub1 : "(P △ Q) << P" by (rule subtypI, erule intE1)
lemma int_sub2 : "(P △ Q) << Q" by (rule subtypI, erule intE2)

lemma subtyp_fun_dom : "R << P ⟹ f : P → Q ⟹ f : R → Q"
proof (rule funI)
  assume "f : P → Q" "R << P"
  fix x assume "x : R" hence "x : P" by (rule subtypE[OF ‹R << P›])
  thus "f x : Q" by (rule funE[OF ‹f : P → Q›])
qed

lemma subtyp_fun_ran : "Q << R ⟹ f : P → Q ⟹ f : P → R"
proof (rule funI)
  assume "f : P → Q" "Q << R"
  fix x assume "x : P" hence "f x : Q" by (rule funE[OF ‹f : P → Q›])
  thus "f x : R" by (rule subtypE[OF ‹Q << R›])
qed

fun mk_styping_trm trm typ styp = 
    ((Const ("Soft_Types.has_ty", typ --> fastype_of styp --> @{typ bool})) $ trm $ styp) 

fun unfold_fun_typing trm =  
  let
    fun funE (f,ty1,ty2) =
      let 
        val x_typ = domain_type (fastype_of f)
        val fx_typ = range_type (fastype_of f)
        val x = Free ("x", x_typ)
        val typ_trm1 = mk_styping_trm x (x_typ) ty1
        val typ_trm2 = (mk_styping_trm (f $ x) (fx_typ) ty2)
      in 
        HOLogic.mk_all ("x", x_typ, 
          HOLogic.mk_imp (unfold_fun_typing typ_trm1, unfold_fun_typing typ_trm2))
      end
    fun depfunE (f, ty1, ty2) = 
      let 
        val x_typ  = domain_type (fastype_of f)
        val fx_typ = range_type (fastype_of f)
        val x = Free ("x", x_typ)
        val typ_trm1 = mk_styping_trm x x_typ ty1
        val typ_trm2 = mk_styping_trm (f $ x) fx_typ (betapply (ty2, x))
      in 
        HOLogic.mk_all ("x", x_typ, 
          HOLogic.mk_imp (unfold_fun_typing typ_trm1, unfold_fun_typing typ_trm2))  
      end
   in
     case trm of
      ((Const ("Soft_Types.has_ty", _)) $ t $ styp) =>
        (case (t,styp) of
          (f, ((Const ("Soft_Types.fun_ty",_)) $ p $ q)) => funE (f, p, q)
        | (f, ((Const ("Soft_Types.depfun_ty",_)) $ p $ q)) => depfunE (f, p, q)
        | _ => trm)
     | _ => trm 
   end

fun is_typing ((Const ("Soft_Types.has_ty",_)) $ _ $ _) = true
  | is_typing _ = false

fun is_fun_ty ((Const ("Soft_Types.fun_ty",_)) $ _ $ _) = true
  | is_fun_ty _ = false

fun is_depfun_ty ((Const ("Soft_Types.depfun_ty",_)) $ _ $ _) = true
  | is_depfun_ty _ = false

fun split_typings ((Const ("Soft_Types.has_ty",_)) $ c $ typ) = (c,typ)
  | split_typings _ = error "not a soft typing"

fun split_fun_ty ((Const ("Soft_Types.fun_ty",_)) $ t $ s) = (t,s)
  | split_fun_ty _ = error "not a function soft type"

fun is_fun_typing trm = is_typing trm andalso 
    (is_fun_ty (snd (split_typings trm)) orelse is_depfun_ty (snd (split_typings trm))) 

fun get_styps ((Const ("Soft_Types.fun_ty",_)) $ t $ s) = t :: get_styps s
  | get_styps t = [t] 

(*Let B be a term, and τ be the type of B, and P be a soft-type on τ:
  Then ‹mk_styping_trm B τ P› and returns the term ‹B : P›*)


fun tall_const T = 
  Const ("Soft_Types.tall", (T --> @{typ bool}) --> 
                 (T --> @{typ bool}) --> @{typ bool});

fun mk_tall (x, T, P, Q) = tall_const T $ P $ absfree (x, T) Q;
