Theory Tagging

theory Tagging
  imports "../GZF/SetComprehension" "../OPair/CartProd"             
          "../Ordinal/Ordinal" "../Ordinal/OrdRec" 
          "../Ordinal/OrdQuants" "../Ordinal/DependentOrd"
begin

class Tagging = OrdRec + CartProd + DepSumOrd

context Tagging begin

subsection ‹Soft typing rules›

definition Tag :: ‹'a ⇒ bool›
  where "Tag ≡ Ord ¦ (λi. i < ω)" 

lemma tagI :
  assumes "i : Ord" "i < ω"
    shows "i : Tag"
  unfolding Tag_def 
  using assms 
  by unfold_typs   

lemma tagD :
  assumes "i : Tag"
  shows "i : Ord" "i < ω"
  using assms
  unfolding Tag_def
  by unfold_typs

lemmas tag_ord = tagD(1)

lemma depfun_ord_setof_set :
  "X : (Π i : Tag. SetOf (α i)) ⟹ X : (Π i : Tag. Set)"
  using setof_set[OF depfunE] by (rule depfunI)

lemma depfun_ord_set_setof : 
  "X : (Π i : Tag. Set) ⟹ X : (Π i : Tag. SetOf (MemOf (X i)))"
  using set_setof_memof[OF depfunE] by (rule depfunI) 

lemmas ord_pmem = smem_pmem[OF ord_setmem]

subsection ‹Tag all members of a set›

definition TagMap :: "'a ⇒ 'a ⇒ 'a" (infix ‹⊕› 60) 
  where "t ⊕ x ≡ { <t, b> | b ∈ x }"

lemma tmap_typ : 
  "TagMap : (α ¦ PairMem) → SetOf β → SetOf (α * β)"
  unfolding TagMap_def
proof (rule funI, drule intE, rule funI, 
       rule repfun_setof[OF setof_set mem_funI], auto)
  fix t b x 
  assume "t : α" "t : PairMem" "x : SetOf β" "b ∈ x" 
  thus "<t,b> : α * β" 
    using productI[OF pair_pair[OF _ smem_pmem]]
          setmemI[OF setof_set] setof_mem by blast
qed

lemmas tmap_setof = funE[OF funE[OF tmap_typ]]

lemma tmap_typ_setof_pair : "TagMap : PairMem → Set → SetOf Pair"
proof (rule funI[OF funI])
  fix t x assume t:"t : PairMem" and x:"x : Set" 
  hence "t : (PairMem ¦ PairMem)" and "x : SetOf SetMem" 
    using intI[OF t] setofI[OF x] setmemI[OF x] by auto
  hence "t ⊕ x : SetOf (PairMem * SetMem)" using tmap_setof by auto
  hence t:"t ⊕ x : Set" and p:"⋀p. p ∈ t ⊕ x ⟹ p : PairMem * SetMem" 
    by (rule setof_set, auto intro: setof_mem) 
  thus "t ⊕ x : SetOf Pair" using setofI[OF t prod_pair[OF p]] by auto 
qed

lemmas tmap_setof_pair = funE[OF funE[OF tmap_typ_setof_pair]]
lemmas tmap_set = setof_set[OF tmap_setof_pair]

lemma tag_setfun : 
  assumes t:"t : PairMem" and x:"x : Set"
    shows "(λb. <t,b>) : x ↝ SetMem"
  by (rule rkpair1[OF x t])

lemma tmap_iff :
  assumes t:"t : PairMem" and x:"x : Set" and b:"b : SetMem"
    shows "b ∈ x ⟷ <t,b> ∈ t ⊕ x"
  unfolding TagMap_def
  by (simp add: repfun_iff[OF x],
      use pair_setmem[OF t smem_pmem[OF b]]
          pair_inject[OF pair_pair[OF t smem_pmem[OF b]] 
                         pair_pair[OF t smem_pmem[OF setmemI[OF x]]]] in auto)

lemma tmapI : 
  assumes t:"t : PairMem" and x:"x : Set"
    shows "b ∈ x ⟹ <t,b> : Pair ∧ <t,b> ∈ t ⊕ x"
  using tmap_iff[OF t x setmemI[OF x]] 
        pair_pair[OF t smem_pmem[OF setmemI[OF x]]] by auto 

lemma tmapD : 
  assumes t:"t : PairMem" and x:"x : Set"
    shows "b ∈ t ⊕ x ⟹ b : Pair ∧ (∃b' ∈ x. b = <t, b'>)"
  unfolding TagMap_def
  by (erule repfunE[OF x], 
      use pair_pair[OF t smem_pmem[OF setmemI[OF x]]] in auto)

lemma tmapE : 
  assumes t:"t : PairMem" and x:"x : Set"
      and b:"b ∈ t ⊕ x"
    obtains b' where "b : Pair" "b' ∈ x" "b = <t, b'>" 
  using tmapD[OF t x b] by auto


subsection ‹Disjoint union of an ordinal-indexed sequence of sets›

definition disj :: "['a ⇒ 'a] ⇒ 'a" (‹⨄›)
  where "disj B ≡ ⋃ i < ω. i ⊕ B i" 

lemma tmap_replfun :
  assumes X:"X : (Π i : Tag. Set)"
  shows "(λi. i ⊕ X i) : predSet ω ↝ Set"
proof (rule mem_funI)
  fix i assume "i ∈ predSet ω"
  hence "i : Ord" "i < ω" 
    using setof_mem[OF predset_setof] 
          predsetE omega_ord 
    by auto
  hence "i : PairMem" "X i : Set" 
    using smem_pmem[OF ord_setmem] depfunE[OF X tagI]
    by auto
  thus "i ⊕ X i : Set" 
    by (rule tmap_set)
qed
 
lemma disj_typ :
   "disj : (Π i : Tag. SetOf (α i)) → SetOf (Σ i : Tag. α i)"
proof (rule funI)
  fix X 
  assume X:"X : (Π i : Tag. SetOf (α i))"
  have X':"X : (Π i : Tag. Set)" by (rule depfun_ord_setof_set[OF X])
  
  show "disj X : SetOf (Σ i : Tag. α i)" unfolding disj_def
  proof (rule setofI[OF union_set[OF repfun_setof[OF 
                predset_set[OF omega_ord] tmap_replfun[OF X']]]])
    fix b assume "b ∈ (⋃ i<ω. i ⊕ X i)"
    then obtain i where 
      i:"i : Ord" and lt:"i < ω" and b:"b ∈ i ⊕ X i"  
      unfolding repfun_union[OF predset_set[OF omega_ord] tmap_replfun[OF X']]
      using predsetE[OF omega_ord] predset_mem_ord[OF omega_ord] by auto
    hence i_pmem:"i : PairMem" and Xi:"X i : SetOf (α i)" 
      using ord_pmem depfunE[OF X tagI] by auto
    then obtain b' where b':"b' ∈ X i" and b':"b = <i, b'>"
      using tmapE[OF _ setof_set b] by auto
    hence "b' : α i" using setof_mem[OF Xi] by auto
    thus "b : (Σ i : Tag. α i)" 
      using depsumI[OF _ tagI[OF i lt]] b' 
            setof_mem[OF tmap_setof_pair[OF i_pmem setof_set[OF Xi]] b] 
      by auto
  qed
qed

lemmas disj_setof = funE[OF disj_typ]

lemma disj_typ_set : "disj : (Π i : Tag. Set) → Set"
proof (rule funI)
  fix X assume X:"X : (Π i : Tag. Set)"
  have "X : (Π i : Tag. SetOf SetMem)" 
    by (rule depfunI, use set_setof[OF depfunE[OF X]] in auto)
  thus "⨄ X : Set" by (rule setof_set[OF disj_setof])
qed

lemmas disj_set = funE[OF disj_typ_set]

lemma tex_oex : 
  assumes j:"j : Ord" 
  shows "(∃ i∈predSet j. P i) ⟷ (∃ i<j. P i)"
  using j predset_iff predset_mem_ord[OF j] by auto

lemma disj_iff : 
  assumes X:"X : (Π i : Tag. Set)" 
  shows "b ∈ ⨄ X ⟷ (∃i : Tag. ∃b'∈ X i. b : Pair ∧ b = <i,b'>)" 
  using tmapD tmapI ord_pmem depfunE[OF X tagI]
  unfolding disj_def repfun_union[OF predset_set[OF omega_ord] 
            tmap_replfun[OF X]]
            tex_oex[OF omega_ord] tex_def Tag_def inter_ty_def has_ty_def by blast
  
(*Weaker version*)
lemma disj_iff' : 
  assumes X:"X : (Π i : Tag. Set)" 
  shows "<i,b'> ∈ ⨄ X ⟷ <i,b'> : Pair ∧ i : Tag ∧ b' ∈ X i"
proof (auto) 
  assume ib:"<i,b'> ∈ ⨄ X"
  thus "<i,b'> : Pair" "i : Tag" "b' ∈ X i"
    using depsumD_pair(1,2,3)[OF setof_mem[OF disj_setof[where α = "λi. MemOf (X i)", 
                                    OF depfun_ord_set_setof[OF X]] ib]] memofD 
    by auto
next
  assume "<i,b'> : Pair" "i : Tag" "b' ∈ X i"
  hence "(∃i' : Tag. ∃c' ∈ X i'. <i,b'> : Pair ∧ <i,b'> = <i',c'>)" by auto
  thus "<i,b'> ∈ ⨄ X" using disj_iff[OF X] by auto
qed


lemma disjI : 
  assumes X:"X : (Π i : Tag. Set)"  
      and ib:"b : Pair" "b = <i,b'>" 
      and ij:"i : Tag" and b':"b' ∈ X i"  
  shows "b ∈ ⨄ X"
  using ib ij b' unfolding disj_iff[OF X] by auto

lemma disjI_pair :
  assumes X:"X : (Π i : Tag. Set)"  
      and ib:"<i,b'> : Pair" 
      and ij:"i : Tag" and b':"b' ∈ X i"  
    shows "<i,b'> ∈ ⨄ X"
  using ib ij b' unfolding disj_iff[OF X] by auto


lemma disjE : 
  assumes X:"X : (Π i : Tag. Set)" 
      and b:"b ∈ ⨄ X"
   obtains i b' where "b : Pair" "b = <i,b'>" "i : Tag" "b' ∈ X i"
  using b unfolding disj_iff[OF X] by auto

lemma disjD_pair : 
  assumes X:"X : (Π i : Tag. Set)"
      and ib:"<i,b'> ∈ ⨄ X"
    shows "i : Tag" "b' ∈ X i"
  using disj_iff'[OF X] ib by auto


definition Part :: "'a ⇒ ('a ⇒ bool)" (‹▸›)
  where "▸ i ≡ (λj. j = i) * Any"

lemma partI_pair : 
  assumes ib:"<i,b'> : Pair"  
    shows "<i,b'> :▸ i"
  unfolding Part_def 
  by (rule productI[OF ib tyI anyI], rule refl)

lemma partI : 
  assumes eq:"b = <i,b'>" and b:"b : Pair"  
    shows "b :▸ i"
  using partI_pair assms by auto

lemma partE :
  assumes "b :▸ i"
  obtains b' where "b : Pair" "b = <i,b'>"
  using assms unfolding Part_def
  by (rule productE, unfold_typs)

lemma partD :
  assumes "b :▸ i"  
    shows "b : Pair" "τ b = i"
  by (rule partE[OF assms], auto, 
      rule partE[OF assms], use fst_eq in auto)

lemma partD_pair :
  assumes "<i,b'> :▸ j"  
  shows "i = j"
  using partD[OF assms] fst_eq by auto

subsection ‹Parts of dependent sums›

lemma depsum_partI :
  assumes ib:"b : (Σ i : Tag. α i)"
  obtains i where "b :▸ i"  
  by (metis depsumE[OF ib] partI_pair)

lemma depsum_partI_pair :
  assumes ib:"<i,b'> : (Σ i : Tag. α i)"
  shows "<i,b'> :▸ i"  
  by (rule partI_pair[OF depsumD(1)[OF ib]])

lemma depsum_partD : 
  assumes b:"b : (Σ i : Tag. α i)" "b :▸ i"
      and i:"i : Tag" 
    shows "π b : α i"
  using depsumD[OF b(1)] partD[OF b(2)] by auto

lemma depsum_partE : 
  assumes i:"i : Tag" 
      and b:"b : (Σ i : Tag. α i)" "b :▸ i"
    obtains b' where "b' : α i"
  using depsumD[OF b(1)] partD[OF b(2)] by auto

lemma depsum_part_projs : 
  assumes b:"b : (Σ i : Tag. α i)" "b :▸ i"
    shows "b : Pair" "τ b : Tag" "π b : α (τ b)"
  using depsumD[OF b(1)] partD[OF b(2)] by auto

lemma depsum_part : 
  assumes b:"b : (Σ i : Tag. α i)"
  obtains i b' where "i : Tag" "b :▸ i" "b' : α i"
  using depsumD[OF b] partI[OF pair_proj_eq] by auto 

lemma tmap_setof_part :
  assumes X:"X : Set" and i:"i : Ord"
    shows "i ⊕ X : SetOf (▸ i)"
  by (unfold Part_def, rule tmap_setof, 
      rule intI[OF tyI ord_pmem[OF i]], auto,
      rule setofI[OF X anyI])

lemma disj_partI :
  assumes X:"X : (Π i : Tag. SetOf (α i))" 
      and b:"b ∈ ⨄ X"
    obtains i where "b :▸ i"
  by (rule depsum_partI[OF setof_mem[OF disj_setof[OF X] b]])

lemma disj_partI_pair :
  assumes X:"X : (Π i : Tag. SetOf (α i))" 
      and b:"<i,b'> ∈ ⨄ X "
    shows "<i,b'> :▸ i"
  by (rule depsum_partI_pair[OF setof_mem[OF disj_setof[OF X] b]])

lemma disj_partE :
  assumes X:"X : (Π i : Tag. SetOf (α i))" 
      and b:"b ∈ ⨄ X" "b :▸ i"
    obtains b' 
      where "i : Tag" "b : Pair" "b = <i,b'>" "b' : α i"
  using depsum_part_projs[OF setof_mem[OF disj_setof[OF X] b(1)] b(2)] 
        pair_proj_eq partD[OF b(2)] by metis

lemma disj_part :
  assumes X:"X : (Π i : Tag. SetOf (α i))" 
      and i:"i : Tag"
      and b:"b = <i,b'>" and b':"b' ∈ X i"
    shows "b :▸ i" "b ∈ ⨄ X"
  by (rule_tac [1] partI[OF b], rule_tac [2] disjI[OF depfun_ord_setof_set[OF X] _ b i(1) b'],
       use b pair_pair[OF ord_pmem[OF tagD(1)[OF i]] 
          smem_pmem[OF setmemI[OF setof_set[OF depfunE[OF X i]] b']]] 
       in auto)

end
