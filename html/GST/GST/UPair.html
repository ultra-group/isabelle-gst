Theory UPair

theory UPair (*Possibly rename to 'FiniteSets'*)
  imports EmptySet
begin

context GZF begin

subsection ‹Unordered pairs; defined by replacement over 𝒫 (𝒫 ∅)›

abbreviation "φ x y a b ≡ (a = ∅ ∧ b = x) ∨ (a = 𝒫 ∅ ∧ b = y)"

definition upair :: "'a ⇒ 'a ⇒ 'a" where
  "upair x y ≡ Repl (𝒫 (𝒫 ∅)) (φ x y)"

definition sng :: "'a ⇒ 'a" where
  "sng x ≡ upair x x"

end

syntax
  "_upair" :: "['d, 'd] ⇒ 'd" (‹{_,_}›)
  "_sng"   :: "'d ⇒ 'd"       (‹{_}›)
translations
  "{X,Y}" ⇌ "CONST upair X Y"
  "{X}"   ⇌ "CONST sng X" 

context GZF begin

lemma pow_not_emp : 
  assumes "x : Set" 
    shows "𝒫 x ≠ ∅"
  by (rule not_emptyI, rule pow_bottom[OF assms])

lemma pemp_set  : "𝒫 ∅ : Set"   by (rule pow_set[OF emp_set])
lemma ppemp_set : "𝒫 𝒫 ∅ : Set" by (rule pow_set[OF pemp_set])

lemma phi_replpred : 
  assumes "x : SetMem" "y : SetMem" "z : Set"
    shows "φ x y : ReplPred z"
proof (insert assms, unfold ReplPred_def has_ty_def, auto)
  fix a show "∃⇩≤⇩1 b : SetMem. φ x y a b" unfolding ReplPred_def 
    by (unfold tuniq_def, rule, use pow_not_emp[OF emp_set] in auto)
qed

lemma upair_typ : "upair : SetMem → SetMem → Set"
proof (rule funI[THEN funI])
  fix a b :: 'a assume ab: "a : SetMem" "b : SetMem" 
  have "𝒫 (𝒫 ∅) : Set" by (rule pow_set[OF pow_set[OF emp_set]])
  thus "{a, b} : Set" unfolding upair_def 
    using repl_set[OF _ phi_replpred[OF ab]] by auto
qed

lemmas upair_set = funE[OF funE[OF upair_typ]]

lemma pow_emp_iff :  "x ∈ 𝒫 ∅ ⟷ x = ∅" 
proof
  assume "x ∈ 𝒫 ∅" 
  hence "x : Set" "x ⊆ ∅" 
    using pow_mem[OF emp_set] pow_iff[OF _ emp_set] by auto
  hence "∀a. a ∉ x" using subset_iff by auto
  thus "x = ∅" using equals0I[OF ‹x : Set›] by blast
next
  assume "x = ∅" thus "x ∈ 𝒫 ∅" 
    using pow_bottom[OF emp_set] by auto
qed

lemma pow_pow_emp_iff : "x ∈ 𝒫 (𝒫 ∅) ⟷ (x = ∅ ∨ x = 𝒫 ∅)" 
proof
  assume "x ∈ 𝒫 𝒫 ∅" hence "x : Set" "x ⊆ 𝒫 ∅" using pow_mem powD pemp_set by auto  
  have "x = ∅ ∨ (∃y. y ∈ x)" by (rule set_disj[OF ‹x : Set›]) 
  thus "x = ∅ ∨ x = 𝒫 ∅"
  proof (rule, simp)
    assume "∃y. y ∈ x" hence "∅ ∈ x" 
      using subsetD[OF ‹x ⊆ 𝒫 ∅›] pow_emp_iff by auto
    have "x = 𝒫 ∅" proof (rule equality_iffI[OF ‹x : Set› pemp_set], rule)
      fix a 
      show "a ∈ x ⟹ a ∈ 𝒫 ∅" using subsetD[OF ‹x ⊆ 𝒫 ∅›] by auto
      show "a ∈ 𝒫 ∅ ⟹ a ∈ x" using ‹∅ ∈ x› pow_emp_iff by auto
    qed
    thus "x = ∅ ∨ x = 𝒫 ∅" by simp
  qed
next
  assume "x = ∅ ∨ x = 𝒫 ∅" hence "x : Set" "x ⊆ 𝒫 ∅" 
    using empty_subsetI subset_refl emp_set pow_set by auto
  thus "x ∈ 𝒫 𝒫 ∅" using powI[OF _ pemp_set] by auto
qed

lemma φ_uniq : "φ x y a b ⟹ (∀z. φ x y a z ⟶ z = b)"
  using pow_emp_iff by blast

lemma upair_iff : 
  assumes "x : SetMem" "y : SetMem" 
  shows "b ∈ upair x y ⟷ (b = x ∨ b = y)"
proof - 
  have pp_emp_set : "𝒫 𝒫 ∅ : Set" by (rule pow_set[OF pow_set[OF emp_set]])
  show ?thesis unfolding upair_def 
  proof (rule, erule replaceE[OF pp_emp_set phi_replpred[OF assms pp_emp_set]])
    fix a assume "a ∈ (𝒫 𝒫 ∅)" and "φ x y a b"
    thus "(b = x ∨ b = y)" using pow_pow_emp_iff by auto
  next
    assume "(b = x ∨ b = y)"
    thus "b ∈ Repl  (𝒫 𝒫 ∅) (φ x y)" 
    proof
      assume "b = x" 
      show "b ∈ Repl (𝒫 𝒫 ∅) (φ x y)"
      proof (rule replaceI[OF pp_emp_set phi_replpred[OF assms pp_emp_set]])
        let ?a = "∅" 
        show "φ x y ?a b" "b : SetMem" 
          using ‹x : SetMem› ‹b = x› by simp_all
        show "∅ ∈ 𝒫 (𝒫 ∅)" 
          using pow_pow_emp_iff by simp
      qed
    next
      assume "b = y" 
      show "b ∈ Repl (𝒫 𝒫 ∅) (φ x y)"
      proof (rule replaceI[OF pp_emp_set phi_replpred[OF assms pp_emp_set]])
        let ?a = "𝒫 ∅" 
        show "φ x y ?a b" "b : SetMem"
          using ‹y : SetMem› ‹b = y› by simp_all
        show "𝒫 ∅ ∈ 𝒫 (𝒫 ∅)" 
          using pow_pow_emp_iff by simp
      qed
    qed
  qed
qed

lemma upairI1 : 
  assumes "a : SetMem" "b : SetMem" 
  shows "a ∈ {a, b}" using upair_iff[OF assms] by simp

lemma upairI2 : 
  assumes "a : SetMem" "b : SetMem"
  shows "b ∈ {a, b}" using upair_iff[OF assms] by simp

lemma upairE : assumes "b : SetMem" "c : SetMem" 
  shows "⟦ a ∈ {b, c} ; a=b ⟹ P ; a=c ⟹ P⟧ ⟹ P" 
  by (simp add: upair_iff[OF assms], blast)


subsection ‹Rules for singleton sets›

lemma sng_typ : "sng : SetMem → Set" 
  by (rule funI, unfold sng_def, rule upair_set, auto) 
lemmas sng_set = funE[OF sng_typ]

lemma sng_iff : 
  assumes "x : SetMem" 
    shows "a ∈ {x} ⟷ a = x"
  unfolding sng_def using upair_iff[OF assms assms] 
  by auto

lemma sngI : 
  assumes "a : SetMem" 
    shows "a ∈ {a}" 
  using sng_iff[OF ‹a : SetMem›] 
  by auto

lemma sngE : 
  assumes "x : SetMem" 
    shows "⟦ a ∈ {x} ; a = x ⟹ P ⟧ ⟹ P"
  using sng_iff[OF ‹x : SetMem›] 
  by auto


subsection ‹Properties of upairs and singletons›

lemma upair_subset: 
  assumes "a : SetMem" "b : SetMem" "x : Set"
    shows "a ∈ x ⟹ b ∈ x ⟹ {a, b} ⊆ x" 
  using upair_iff[OF assms(1,2)] 
  by auto

lemma sng_subset: 
  assumes "a : SetMem" "x : Set" 
    shows "a ∈ x ⟹ {a} ⊆ x" 
  unfolding sng_def 
  using upair_subset[OF ‹a : SetMem› ‹a : SetMem› ‹x : Set›] 
  by auto

lemma sng_eq_iff : 
  assumes "a : SetMem" "b : SetMem"
    shows "{a} = {b} ⟷ a=b"
  by (rule equality_iff[OF sng_set[OF ‹a : SetMem›], OF sng_set[OF ‹b : SetMem›], THEN cnf.iff_trans],
      use sng_iff assms in auto)

lemma upair_eq_iff : 
  assumes "a : SetMem" "b : SetMem" "c : SetMem" "d : SetMem" 
    shows "{a, b} = {c,d} ⟷ ((a=c ∧ b=d) ∨ (a=d ∧ b=c))"
proof (rule equality_iff[OF upair_set[OF assms(1,2)] upair_set[OF assms(3,4)], THEN cnf.iff_trans], rule)
  assume eq:"∀x. x ∈ {a, b} ⟷ x ∈ {c,d}"
  have "a = b ∨ a ≠ b" by simp
  thus "a = c ∧ b = d ∨ a = d ∧ b = c"
  proof
    assume "a = b" hence "{a, b} = {a}" unfolding sng_def by simp
    hence "c = a ∧ d = a" using eq upair_iff assms sng_iff by auto
    thus "a = c ∧ b = d ∨ a = d ∧ b = c" using ‹a = b› by auto
  next
    assume "a ≠ b"
    have "a ∈ upair c d" "b ∈ upair c d" using eq upairI1 upairI2 assms by auto
    thus "a = c ∧ b = d ∨ a = d ∧ b = c" using ‹a≠b› upair_iff assms by auto
  qed
next
  assume "a = c ∧ b = d ∨ a = d ∧ b = c" 
  thus "∀x. (x ∈ {a, b}) ⟷ (x ∈ upair c d)" using upair_iff assms by auto
qed

end
