Theory mlt

theory mlt
  imports mOrd
begin

context Ordinal_Model
begin


lemma mlt_mord : 
  "i ≪ j ⟹ i : mOrd ∧ j : mOrd"
  unfolding mlt_def by simp

lemmas mlt_mord1 = conjunct1[OF mlt_mord]
   and mlt_mord2 = conjunct2[OF mlt_mord]

lemma mltI : 
  assumes i : "i : Ord" and j : "j : Ord"
    shows "i < j ⟹ <ord, i> ≪ <ord, j>"
    unfolding mlt_def
    using mordI i j ord_snd_eq' by auto

lemma mltI_eq : 
  assumes i : "i' : Ord" "i = <ord, i'>" 
      and j : "j' : Ord" "j = <ord, j'>"
    shows "i' < j' ⟹ i ≪ j"
    using mltI assms by auto

lemma mltD :
  assumes "<ord, i> ≪ <ord, j>"
  shows "i < j"
  using assms mord_snd_eq
  unfolding mlt_def by auto

lemma mltE1 :
  assumes ij:"i ≪ j"
  obtains i' j' 
    where "i' : Ord" "i = <ord, i'>" 
          "j' : Ord" "j = <ord, j'>" 
          "i' < j'"
proof -
  obtain i' j' where 
    "i' : Ord" "i = <ord, i'>" "j' : Ord" "j = <ord, j'>" 
    using mordE mlt_mord[OF ij] by metis
  moreover hence "i' < j'" 
    using mltD ij by auto
  ultimately show ?thesis ..
qed

(*Less info than mltE1, maybe more efficient?*)
lemma mltE2 : 
  assumes ij:"i ≪ j"
  obtains i' j' 
    where "i = <ord, i'>" "j = <ord, j'>" "i' < j'"
  using mltE1[OF ij] .

lemma mlt_trans :
  assumes ij: "i ≪ j" and jk: "j ≪ k"
    shows "i ≪ k"
proof (rule mltE1[OF ij], rule mltE1[OF jk]) 
  fix i' j' j'' k' 
  assume i':"i' : Ord" "i = <ord, i'>"
     and "j' : Ord" "j = <ord, j'>"
     and "j'' : Ord" "j = <ord, j''>"
     and k':"k' : Ord" "k = <ord, k'>"
     and "i' < j'" "j'' < k'"
  hence "i' < j'" "j' < k'" 
    using mord_pair_inject by auto
  hence "i' < k'" 
    using trans[OF ‹i' : Ord› ‹j' : Ord› ‹k' : Ord›] by auto
  thus "i ≪ k"
    using mltI_eq[OF i' k'] by auto
qed

theorem mlt_trans_ax :
  "m∀i : mOrd. m∀j : mOrd. m∀k : mOrd. i ≪ j ⟶ j ≪ k ⟶ i ≪ k"
  unfolding mtall_def 
  using mlt_trans by blast

lemma mlt_notsym :
  assumes ij:"i ≪ j"
  shows "¬ j ≪ i"
proof (rule mltE1[OF ij])
  fix i' j' 
  assume i':"i' : Ord" "i = <ord, i'>"
     and j':"j' : Ord" "j = <ord, j'>"
     and ij:"i' < j'"
  show "¬ j ≪ i"
  by (rule asym[OF i'(1) j'(1) ij], 
      unfold i' j', simp add: mltD)     
qed

lemma mlt_notsym_ax : 
  "m∀i : mOrd. m∀j : mOrd. i ≪ j ⟶ ¬ j ≪ i"
  unfolding mtall_def 
  using mlt_notsym by auto

lemma mlt_linear :
  assumes i : "i : mOrd" and j : "j : mOrd"
    shows "i ≪ j ∨ i = j ∨ j ≪ i" 
proof (rule mordE[OF i], rule mordE[OF j])
  fix i' j' 
  assume i':"i' : Ord" "i = <ord, i'>"
     and j':"j' : Ord" "j = <ord, j'>"
  have "i' < j' ∨ i' = j' ∨ j' < i'"
    using linear[OF i'(1) j'(1)] by auto
  thus ?thesis unfolding i' j'
    using mltI[OF i'(1) j'(1)]
          mltI[OF j'(1) i'(1)] by auto
qed

lemma mlt_linear_ax :
  "m∀i : mOrd. m∀j : mOrd. i ≪ j ∨ i = j ∨ j ≪ i"
  unfolding mtall_def
  using mlt_linear by auto
 
lemma mlt_induct :  
  assumes i : "⋀i. ⟦ i : mOrd ; ⋀j. j : mOrd ⟹ j ≪ i ⟹ P j ⟧ ⟹ P i" 
      and a : "a : mOrd"
    shows "P a"
proof (rule mordE[OF a])
  let ?P' = "λb'. P (<ord, b'>)"
  fix a' assume 
    a' : "a' : Ord" "a = <ord, a'>" 
  {
    fix i' 
    assume i' : "i' : Ord" 
       and ih : "⋀j'. j' : Ord ⟹ j' < i' ⟹ ?P' j'"
    hence "<ord, i'> : mOrd" 
      using mordI by auto
    moreover have "⋀j. j : mOrd ⟹ j ≪ <ord,i'> ⟹ P j"
    proof -
      fix j assume "j : mOrd" "j ≪ <ord, i'>"
      then obtain j' where "j' : Ord" "j = <ord, j'>" "j' < i'"
        using mltD mordE by metis
      thus "P j" using ih by auto
    qed
    hence "?P' i'" using i[OF ‹<ord, i'> : mOrd›] by auto
  }
  hence "?P' a'" 
    by (rule trans_induct[OF ‹a' : Ord›, of ?P'])
  thus "P a"
    unfolding a' .
qed

lemma mlt_induct_ax :
  "∀P. (m∀i : mOrd. (m∀j : mOrd. j ≪ i ⟶ P j) ⟶ P i) ⟶ (m∀i : mOrd. P i)"
proof (rule)
  fix P show 
    "(m∀i : mOrd. (m∀j : mOrd. j ≪ i ⟶ P j) ⟶ P i) ⟶ (m∀i : mOrd. P i)"  
    unfolding mall_def mtall_def tall_def
    using mord_m by (metis mlt_induct[of P])
qed

end
end
