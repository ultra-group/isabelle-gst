(*Comments for theories*)

(*Hiding syntax and translations for HOL-set-bounded quantification*)
no_translations
  "∀x∈A. P" ⇌ "CONST Ball A (λx. P)"
  "∃x∈A. P" ⇌ "CONST Bex A (λx. P)"

  "_Ball"       :: "pttrn ⇒ 'a set ⇒ bool ⇒ bool"  ("(3∀(_/∈_)./ _)" [0, 0, 10] 10)
  "_Bex"        :: "pttrn ⇒ 'a set ⇒ bool ⇒ bool"   ("(3∃(_/∈_)./ _)" [0, 0, 10] 10)

(*Hiding syntax for HOL set comprehension notation*)

  "{x. P}" ⇌ "CONST Collect (λx. P)"

  "_Coll" :: "pttrn ⇒ bool ⇒ 'a set"    ("(1{_./ _})")

  "{x, xs}" ⇌ "CONST insert x {xs}"
  "{x}" ⇌ "CONST insert x {}"

  "_Finset" :: "args ⇒ 'a set"    ("{(_)}")

no_translations
  "⋂x∈A. f" ⇌ "CONST Inter ((λx. f) ` A)"
  "⋃x∈A. f" ⇌ "CONST Union ((λx. f) ` A)"
  "⋃i<n. A" ⇌ "⋃i∈{..<n}. A"
no_syntax
  "_INTER"      :: "pttrn ⇒ 'a set ⇒ 'b set ⇒ 'b set"  ("(3⋂_∈_./ _)" [0, 0, 10] 10)
  "_UNION"      :: "pttrn ⇒ 'a set ⇒ 'b set ⇒ 'b set"  ("(3⋃_∈_./ _)" [0, 0, 10] 10)
  "_UNION_less" :: "'a => 'a => 'b set => 'b set"       ("(3⋃_<_./ _)" [0, 0, 10] 10)

(*Hiding HOL set theory notation*)
